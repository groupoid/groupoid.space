\keys_define:nn { jmsdelim } {
  size~commands .clist_set:N = \l__jmsdelim_size_cmds,

  size~commands = {relax,jmsdelim_big:n,jmsdelim_Big:n,jmsdelim_bigg:n,jmsdelim_Bigg:n},
}
\cs_new:Npn \jmsdelim_big:n #1 {
  {\mathchoice{\big #1} {\big #1}{\big #1}{#1}}
}
\cs_new:Npn \jmsdelim_Big:n #1 {
  {\mathchoice{\Big #1} {\Big #1}{\big #1}{#1}}
}
\cs_new:Npn \jmsdelim_bigg:n #1 {
  {\mathchoice{\bigg #1} {\bigg #1}{\big #1}{#1}}
}

  {\mathchoice{\Bigg #1} {\Bigg #1}{\big #1}{#1}}
}
\int_new:N \g__jmsdelim_size
\int_new:N \g__jmsdelim_size_up
\int_gset:Nn \g__jmsdelim_size {0}
\int_gset:Nn \g__jmsdelim_size_up {0}
\cs_new:Npn \__jmsdelim_clist_item:Nn #1 #2 {
  \clist_item:Nn #1 {
    \int_min:nn { #2 } {\clist_count:N #1}
  }
}
\cs_new:Npn \__jmsdelim_setup_sizes: {
  \int_gset:Nn \g__jmsdelim_size {
    \int_max:nn \g__jmsdelim_size \g__jmsdelim_size_up
  }
  \cs_set_eq:Nc \jmsdelim_size_cmd: {
    \__jmsdelim_clist_item:Nn \l__jmsdelim_size_cmds {
      \g__jmsdelim_size + 1
    }
  }
}
\cs_new:Npn \__jmsdelim_luatex_save_mathstyle:N #1 {
  \ifcase \mathstyle
    \cs_set_eq:NN #1 \displaystyle
  \or
    \cs_set_eq:NN #1 \crampeddisplaystyle
  \or
    \cs_set_eq:NN #1 \textstyle
  \or
    \cs_set_eq:NN #1 \crampedtextstyle
  \or
    \cs_set_eq:NN #1 \scriptstyle
  \or
    \cs_set_eq:NN #1 \crampedscriptstyle
  \or
    \cs_set_eq:NN #1 \scriptscriptstyle
  \or
    \cs_set_eq:NN #1 \crampedscriptscriptstyle
  \fi
}
\cs_new:Npn \__jmsdelim_restore_mathstyle: {
  \SavedStyle
}
\cs_new:Npn \__jmsdelim_save_mathstyle:n #1 {
  \ifluatex
    \__jmsdelim_luatex_save_mathstyle:N \__jmsdelim_restore_mathstyle:
    #1
  \else
    \ThisStyle{#1}
  \fi
}
\cs_new:Npn \jmsdelim_scope:nn #1 #2 {
  \group_begin:
    \int_set:Nn \l_tmpa_int \g__jmsdelim_size_up
    \int_gset:Nn \g__jmsdelim_size_up 0
    \int_gset:Nn \g__jmsdelim_size 0
    \group_begin:
      \__jmsdelim_save_mathstyle:n {
        #1
        \__jmsdelim_setup_sizes:
        #2
      }
    \group_end:
    \int_gset:Nn \g__jmsdelim_size_up {\int_max:nn \g__jmsdelim_size_up \l_tmpa_int}
  \group_end:
}

   \mode_if_math:TF

\title{Вкладення Йонеди \\ в сімпліціальній теорії типів}
\author{Данієль Гратцер, Джонатан Вайнбергер та Ульрік Бухгольц}
\maketitle

\begin{abstract}
  Ріль і Шульман \textcite{riehl:2017} ввели \emph{симпліціальну теорію типів} (\STT), варіант гомотопічної теорії
  типів, який мав на меті вивчати не лише теорію гомотопій, але і її злиття з теорією категорій:
  теорію $\prn{\infty,1}$-категорій. Хоча це і є технічно складним, маніпулювання $\infty$-категоріями в
  симпліціальній теорії типів часто є простішим, ніж робота зі звичайними категоріями, оскільки теорія типів
  обробляє нескінченні стеки когерентностей у фоновому режимі. Ми спираємося на нещодавню роботу
  \textcite{gratzer:2024}, що визначає $\prn{\infty,1}$-категорію $\infty$-групоїдів у \STT{}, щоб
  визначити категорії передпучків у межах \STT{} та систематично розвинути їхню теорію. Зокрема,
  ми будуємо вкладення Йонеди, доводимо універсальну властивість категорій передпучків, уточнюємо
  теорію спряженостей у \STT{}, вводимо теорію розширень Кана та доводимо теорему А
  Квіллена. На додаток до великого обсягу теорії категорій у \STT{}, ми пропонуємо вагомі
  докази того, що \STT{} може бути використана для отримання складних результатів у теорії $\infty$-категорій з
  набагато меншою складністю.
\end{abstract}

\section{Вступ}

Расселл \textcite{russell:1919} відомим чином описав два стилі формалізації математики як різницю
між \emph{крадіжкою} та \emph{чесною працею}. Обидва підходи можна побачити в сучасному використанні
залежної теорії типів. Чесна праця передбачає \emph{аналітичний} підхід: трактування типів як
базових об'єктів, еквівалентних множинам, а також визначення і міркування про такі об'єкти, як дійсні числа,
групи та топологічні простори, як це робиться зазвичай. Це те, що робиться, наприклад, у доведенні Coq
Теореми про непарний порядок~\parencite{gonthier:2013}. Більш швидкий шлях крадіжки передбачає
розгляд теорії типів як спеціалізованої \emph{синтетичної} мови для певного типу математичних
об'єктів і постулювання їх базових властивостей. Це звужує сферу застосування теорії типів, але, водночас,
робить доведення про ці конкретні об'єкти набагато лаконічнішими. Наприклад,
\emph{гомотопічна теорія типів} (\HoTT{})~\parencite{hottbook} постулює різні аксіоми, які гарантують, що типи
поводяться як простори (з точністю до гомотопії), уможливлюючи доведення теорем з алгебраїчної топології
без введення явного опису простору. Насправді, синтетичний підхід
менше схожий на крадіжку, ніж на позику; ви розплачуєтеся за налаштовану теорію типів
семантичною моделлю, яка інтерпретує типи як потрібні об'єкти і валідує додаткові
аксіоми.

У цій роботі ми використовуємо синтетичну методологію для застосування теорії типів до вивчення теорії
категорій. Зокрема, ми додаємо різні аксіоми до гомотопічної теорії типів, щоб побудувати систему,
де гасло \HoTT{} ``всі типи є просторами, а всі функції неперервними'' замінюється на
``(деякі) типи є ($\infty$-)категоріями, а всі функції є функторами''.\footnote{%
  У цій статті під $\infty$-категорією ми розуміємо $\prn{\infty,1}$-категорію.}
Це розширення
теорії типів називається \emph{симпліціальною} теорією типів (\STT{}) і було введено
\textcite{riehl:2017}.

Хоча знання $\infty$-категорій не є обов'язковим для використання нашої теорії, груба інтуїція щодо них
допомагає для розуміння \STT{}. Тому ми нагадаємо наступне нечітке визначення.
$\infty$-категорія $C$ -- це сукупність об'єктів з \emph{простором} стрілок між об'єктами $c$
та $d$, $\Hom{c}{d}$, а не множиною, оснащена неперервною операцією композиції та призначенням
тотожних стрілок. Вирішальним є те, що операція композиції має бути асоціативною та унітальною лише з
точністю до гомотопії, але з обмеженням, що ці гомотопії самі задовольняють закони \emph{когерентності} у
вигляді додаткових гомотопій, і так далі з когерентностями між когерентностями тощо. Як вільна
аналогія, подібно до того, як моноїдальна категорія послаблює моноїди, дозволяючи $\otimes$ бути асоціативним з
точністю до ізоморфізмів, що задовольняють певні рівняння когерентності, $\infty$-категорії послаблюють звичайні категорії
дозволяючи законам категорій виконуватися лише з точністю до (нескінченно когерентних) ізоморфізмів.

Прикметно, що практично кожна теорема, на яку можна сподіватися для звичайних категорій, виконується для
$\infty$-категорій.\footnote{Принаймні, як зауважив один з наших рецензентів, за умови правильного калібрування своїх сподівань.} Однак,
доведення є набагато складнішими, оскільки вони працюють з \emph{моделями} $\infty$-категорій (інструментами, що використовуються для
організації та управління вежею когерентностей~\parencite{bergner:2018}). Метою \STT{} є використання
теорії типів, щоб приховати когерентності від користувача і дозволити доведення, які є не складнішими, ніж
класичні міркування для $1$-категорій.

У цій роботі ми надаємо вагомі докази цієї гіпотези, розвиваючи велику частину
теорії категорій---кілька основних результатів \emph{Категорій для працюючого
  математика}~\parencite{maclane:1978}---виключно в межах \STT{}.

\subsection{Симпліціальна теорія типів}

Щоб побудувати теорію типів для синтетичної теорії категорій, можна було б сподіватися інтерпретувати теорію типів у
категорію категорій ($\infty$ або інакше), щоб гарантувати, що типи реалізують категорії. Однак,
категорія $\CAT$ малих категорій поводиться занадто погано, щоб утворити модель теорії типів
Мартіна-Льофа (\MLTT{}). Замість цього, \textcite{riehl:2017} розширюють $\CAT$ і вкладають її як рефлективну
підкатегорію в категорію ($\infty$-)передпучків на симплекс-категорії $\PSH{\SIMP}$, яка є достатньо
багатою для моделювання \HoTT{}. Тоді \STT{} аксіоматизує частину $\PSH{\SIMP}$, щоб ізолювати $\CAT$ як
рефлективний підуніверсум у межах теорії типів~\parencite{rijke:2020}.

Ми введемо повний набір доповнень у Розділі~\ref{sec:prelims} (зібраних у Додатку~\ref{sec:axioms}
для зручності), але найважливішим серед них є постульований \emph{інтервальний тип}
$\Int : \Uni[0]$. Ми додатково припускаємо, що $\Int$ є обмеженим лінійним порядком з кінцевими точками
$0,1 : \Int$. Інтуїтивно, $\Int$ має фіксувати категорію $\brc{0 \to 1}$---він так
інтерпретується в $\PSH{\SIMP}$---і ми можемо використовувати це для визначення та дослідження типу
\emph{синтетичних морфізмів} у довільному типі $X$: стрілка в $X$ відповідає звичайній
функції $\Int \to X$, причому обчислення в $0,1$ дає домен і кодомен. Наприклад,
тотожна стрілка в $x : X$ задається як $\lambda \_.\,x$.

Однак, подібно до того, як цільова модель $\PSH{\SIMP}$ є строго більшою за $\CAT$, не всі типи в \STT{}
достовірно моделюють категорії. Зокрема, хоча завжди можна побудувати тотожні
морфізми, не всі типи мають оператор композиції. Дивовижно, проте, що оператори композиції
є унікальними, коли вони існують, і їх існування для типу $X$ фіксується відносно коротким
твердженням (Визначення~\ref{def:prelims:category}). Маючи під рукою операцію композиції для $X$, ми можемо
визначити тип ізоморфізмів у $X$, і ми визначаємо категорію як тип, де (1)
операція композиції існує унікально з точністю до гомотопії, і (2) тип ізоморфізмів у $X$ еквівалентний типу ідентичності $=_X$.

\begin{rem}
  Цей останній пункт принципово залежить від \emph{відмови} від припущення унікальності доведень ідентичності, щоб ми
  випадково не заборонили будь-якій синтетичній категорії мати об'єкт з нетривіальним
  автоморфізмом. Однак, припускаючи, що ізоморфізми та доведення ідентичності збігаються, ми можемо
  використовувати підтримку теорією типів заміни рівного на рівне для безперешкодного перенесення доведень уздовж
  ізоморфізмів. Ось чому робота з \HoTT{}/інтенсіональною теорією типів при формулюванні синтетичної
  теорії категорій виявляється зручнішою, ніж екстенсіональна теорія типів, навіть якщо хтось не цікавиться
  $\infty$-категоріями.
\end{rem}


\subsection{Теорія категорій всередині \texorpdfstring{\STT{}}{STT}}

Хоча деякі нещодавні роботи досліджували \STT{} для її застосувань у мовах
програмування~\parencite{weaver:2020,gratzer:2024,weaver:phd}, більшість робіт із симпліціальної теорії типів були
зосереджені на доведенні результатів з теорії категорій всередині теорії
типів~\parencite{riehl:2017,riehl:2023,bardomiano:2021,weinberger:phd,buchholtz:2023,weinberger:twosided:2024,weinberger:sums:2024}.
З цією метою теорія спряженостей, дискретних розшарувань та розшарувань Гротендіка, а також (ко)границь були
введені та вивчені в межах симпліціальної теорії типів. Деякі з цих результатів, наприклад,
лема Йонеди для розшарувань~\parencite{riehl:2017}, були згодом
механізовані~\parencite{kudasov:2024}.

Донедавна, однак, у \STT{} не існувало замкнених типів, які б представляли нетривіальні
категорії. Як наслідок, хоча \textcite{riehl:2017} і наводить відмінне визначення спряженостей,
жодних прикладів навести не можна. У попередній роботі ми змінили це,
розширивши \STT{} для побудови $\Space$, категорії $\mathcal{S}$просторів, яка є гомотопічним аналогом $\SET$~\parencite{gratzer:2024}.
Об'єкти $\Space$ -- це елементи $\Uni[0]$, які кодують $\infty$-групоїди, а морфізми в
$\Space$ відповідають їх функціям. \emph{Там само} $\Space$ використовується як будівельний блок для
відновлення алгебраїчних категорій (груп, кілець), а також інших прикладів (частково впорядкованих множин,
симплекс-категорії тощо).

Наше розширення \STT{} використовувало різні \emph{модальності} поверх \HoTT{} для побудови $\Space$. Тут ми беремо $\Space$ повністю, але деякі з використаних нами модальностей все ще є критично важливими для формулювання природних теорем
у теорії категорій. Відповідно, у цій статті ми також працюємо в межах модального розширення \HoTT{}, заснованого на
\MTT{}~\parencite{gratzer:2020}.

\subsection{Внесок}
Ми переглядаємо базову теорію категорій у світлі побудови $\Space$ і показуємо, що
більшість класичних результатів, з якими стикаються в теорії категорій, тепер знаходяться в межах досяжності симпліціальної
теорії типів. Вперше ми показуємо, що \STT{} може бути використана для доведення життєво важливих теорем у
теорії $\infty$-категорій без звернення до складних моделей. Багато з цих теорем
(наприклад, цілком лояльні та суттєво сюр'єктивні функтори є еквівалентностями) прямо не згадують
$\Space$, але критично спираються на принципи міркування, які уможливлює $\Space$. Ми доводимо два
робочі результати для категорій передпучків $\PSH{C}$:
\begin{itemize}
\item Ми будуємо цілком лояльну функцію $\Yo : C \to \PSH{C}$.
\item Ми доводимо, що $\PSH{C}$ є ``вільним копоповненням $C$''.
\end{itemize}
Ключовою технічною інновацією для цього є категорія \emph{скручених стрілок} (twisted arrow), яку ми інтегруємо в
\STT{} як модальність. Після цього ми можемо вивести різні класичні результати, наприклад:
\begin{itemize}
\item що поточково оборотні відображення в $C \to D$ є оборотними;
\item що поточково ліві спряжені є лівими спряженими;
\item що (ко)границі обчислюються поточково в $C \to D$;
\item теорію та існування поточкових розширень Кана;
\item теорему А Квіллена;
\item власніть кодекартових розшарувань.
\end{itemize}

Синтетичний підхід забезпечує стислі доведення для багатьох із цих теорем порівняно з класичним викладом у теорії $1$-категорій, однак наші доведення застосовні також і до $\infty$-категорій, де покращення є набагато більш радикальними: у \textcite{lurie:2009} доведення того, що $\Yo$ є цілком лояльним, займає сотні сторінок, а доведення того, що поточково природні перетворення є ізоморфізмами, вимагає майже п'яти сторінок зусиль у \textcite{cisinski:2019}. Розподіляючи роботу між конструкцією всередині \STT{} та вже існуючою моделлю \STT{}, ми уникаємо багатьох із цих технічних складнощів та наводимо доведення, більш звичні для спеціалістів із теорії $1$-категорій. Зокрема, ми показуємо, що подібно до того, як гомотопічна теорія типів дозволила теоретикам типів отримати нові результати в алгебраїчній топології, симпліціальна теорія типів дає їм змогу зробити те саме в теорії $\infty$-категорій.

\begin{rem}
  Оскільки \STT{} розширює \HoTT{} низкою аксіом, природно запитати, чи є ці аксіоми в якомусь сенсі \emph{повними}. Наш теперішній набір аксіом не є повним для запланованих моделей симпліціальних об'єктів в $\infty$-топосі (хоча вони є несуперечливими), але це не є ні дивним, ні небажаним: сама \HoTT{} не є повною для своїх запланованих моделей ($\infty$-топосів), а її екзотичні моделі становлять значний інтерес. Аналогічно, ми очікуємо, що \STT{} матиме цікаві екзотичні моделі, і не можемо обґрунтовано сподіватися на те, що скінченний набір аксіом буде повним для стандартних моделей. Набагато важливішим є те, чи достатньо цих аксіом для виведення стандартних результатів теорії категорій, що є емпіричним, а не суто математичним питанням. Дійсно, у споріднених синтетичних підходах до теорії доменів~\parencite{hyland:1991}, диференціальної геометрії~\parencite{kock:2006} та алгебраїчної геометрії~\parencite{cherubini:2023} точні аксіоми формувалися протягом кількох років та багатьох ітерацій. З огляду на це, ми вважаємо, що наші результати дають вагомі свідчення на користь виражальної сили цього набору аксіом.
\end{rem}

\subsection{Структура статті}
У Розділі~\ref{sec:prelims} ми оглядаємо ключові аспекти, що становлять основу цієї роботи: гомотопічну теорію типів, базову симпліціальну теорію типів, модальну гомотопічну теорію типів та їхній синтез: \STT{}. У Розділі~\ref{sec:tw} ми вивчаємо категорію \emph{скручених стрілок} і використовуємо її для побудови вкладення Йонеди. Ми доводимо кілька дедалі складніших версій леми Йонеди й завершуємо повністю функторіальною версією (Теорема~\ref{thm:tw:strong-yoneda}). У Розділі~\ref{sec:adj} ми застосовуємо лему Йонеди для перегляду теорії спряженостей, запропонованої \textcite{riehl:2017}. Ми розробляємо кілька інструментів для побудови спряженостей і використовуємо їх, щоб навести перші нетривіальні приклади спряженостей у \STT{}. Ми також використовуємо цей апарат, щоб показати, що $\PSH{C}$ є вільним копоповненням $C$ (Теорема~\ref{thm:adj:cc}). У Розділі~\ref{sec:kan} ми розвиваємо теорію розширень Кана в \STT{} і доводимо кілька важливих результатів: існування поточкових розширень Кана (Теорема~\ref{thm:kan:left-kan-exists}), теорему А Квіллена (Теорема~\ref{thm:kan:theorem-a}) та властивість власності кодекартових відображень (Теорема~\ref{thm:kan:cocartesian-is-proper}). Наше доведення останнього факту є особливо примітним, оскільки застосування теорії типів дозволило нам знайти набагато простіше доведення, ніж відомі нам з літератури.

\section{Модальна та симпліціальна теорія типів}
\label{sec:prelims}

У цій статті ми сприймаємо \STT{} значною мірою як належне і зосереджуємося на роботі всередині теорії. Однак, щоб зробити цю статтю більш самодостатньою, ми присвячуємо цей розділ ретельному поясненню нових конструкцій модальної гомотопічної теорії типів та аксіом, що її доповнюють і утворюють симпліціальну теорію типів.

\subsection{Гомотопічна теорія типів}
Ми починаємо з нагадування основних понять і позначень гомотопічної теорії типів, які ми використовуємо в цій статті. Канонічним джерелом є книга \HoTT{}~\parencite{hottbook}. Ми працюємо в межах інтенсіональної теорії типів Мартіна-Льофа і зауважимо, як \HoTT{} розширює її.

\begin{nota}
  Ми пишемо $a =_A b$ для типу ідентичності (часто опускаючи $A$). Дано $p : a =_A b$ та
  $B : A \to \Uni$, ми пишемо $p_!$ для відображення $B\prn{a} \to B\prn{b}$.
\end{nota}

\begin{defi}
  Ми кажемо, що функція $f : A \to B$ є еквівалентністю, якщо $f$ допускає як лівий, так і правий
  обернений:
  \[
    \IsEquiv\prn{f} =
    \Sum{g,h : B \to A} \prn{g \circ f = \ArrId{}} \times \prn{f \circ h = \ArrId{}}
  \]
  Ми пишемо $A \Equiv B$ для суми $\Sum{f : A \to B} \IsEquiv\prn{f}$.
\end{defi}
\HoTT{} є розширенням інтенсіональної теорії типів з ієрархією універсумів, що задовольняють аксіому унівалентності:
\[
  \Con{univ}_i
  :
  \Prod{A, B : \Uni[i]} \IsEquiv\prn{
    \lambda p.\,\prn{p_!, \cdots} : A =_{\Uni[i]} B \to A \Equiv B
  }
\]
Ми будемо опускати $i$ в $\Con{univ}_i$ та $\Uni[i]$ і ігнорувати питання розміру, якщо вони не є суттєвими. Унівалентність породжує велику кількість шляхів у $\Uni$, які відрізняються від $\Refl$. Нас часто цікавлять типи, які є тривіальними, мають лише тривіальні шляхи або тривіальні шляхи між шляхами і так далі. Ці умови організовані в сімейство предикатів, що називаються рівнем усічення ($-2,-1,0,\ldots$) типу. Ми будемо використовувати лише перші три рівні, стверджуючи, що тип є стягуваним, (гомотопічною) пропозицією або множиною:
\begin{mathparpagebreakable}
  \IsContr\prn{A} = \Sum{a : A} \Prod{b : A} a = b
  \and
  \IsProp\prn{A} = \Prod{a,b : A} \IsContr\prn{a = b}
  \and
  \IsSet\prn{A} = \Prod{a,b : A} \IsProp\prn{a = b}
\end{mathparpagebreakable}

\begin{prop}[\textcite{shulman:2019}, див. також \textcite{riehl:2024}]
  \label{thm:prelims:hott-models}
  Усі теоретико-типові модельні топоси
  \textup(і, отже, $\infty$-топоси Гротендіка\textup) моделюють \HoTT{}.
\end{prop}

Ми також матимемо нагоду використовувати різні \emph{вищі індуктивні типи} (ВІТ). Семантика ВІТ є складною і безпосередньо не розглядається наведеним вище результатом~\parencite{lumsdaine:2019}. Зокрема, хоча \textcite{shulman:2019} показує, що вищезгадана модель підтримує всі вищі індуктивні типи, він не показує, що універсуми є строго замкненими щодо цих конструкцій. Хоча робота над отриманням цього результату триває, легко показати, що універсуми є \emph{слабко} замкненими щодо цих конструкцій. Наприклад, існує тип $D : \Uni[0]$, такий що $D \Equiv A \Pushout{C} B$ щоразу, коли $A,B,C : \Uni[0]$. Відповідно, ми будемо припускати, що наші універсуми замкнені щодо вищих індуктивних типів, хоча і лише з пропозиційними $\beta$-правилами.

\subsection{Симпліціальна теорія типів}
Маючи під рукою \HoTT{}, ми переходимо до симпліціальної теорії типів. Це розширення \HoTT{} кількома аксіомами, які дозволяють нам розглядати (певні) типи як $(\infty,1)$-категорії, що надалі будуть називатися просто категоріями. Ми, відповідно, опускатимемо префікс $(\infty,1)$- або $\infty$- скрізь. По-перше і найфундаментальніше, ми додаємо наступне:

\begin{restatable}{axiom}{intaxiom}
  \label{ax:int}
  Існує множина $\Int$, що утворює обмежену дистрибутивну ґратку $\prn{0,1,\lor,\land}$ таку,
  що $\Prod{i,j : \Int} i \le j \lor j \le i$ виконується.
\end{restatable}
Ми розглядаємо $\Int$ як \emph{напрямлений інтервал}, і \textcite{riehl:2017} використовують це, щоб надати кожному типу поняття синтетичного морфізму:
\begin{defi}
  Синтетичний морфізм $f : \Hom[X]{x}{y}$ між $x,y : X$ - це функція $f : \Int \to X$ разом
  з пропозиційними рівностями $f\,0 =_X x$ та $f\,1 =_X y$.
\end{defi}

\begin{rem}
  В \textcite{riehl:2017}, синтетичний інтервал визначається як більш примітивна судженнява структура
  і $\Hom{x}{y}$ використовує строгі типи розширення. Це дає більше визначених рівностей: $f\,0$ та
  $x$ співпадатимуть за означенням, коли $f : \Hom{x}{y}$ (і аналогічно для $f\,1$ та $y$). Проте, підхід
  суджень не дозволяє прямо включити $\Int$ як звичайний тип, і його взаємодії з модальностями
  (Розділ~\ref{sec:prelims:modal-stt}) є складними. З цих причин, ми працюємо з простішим, але менш
  строгим означенням $\Hom{x}{y}$. В системі, де обидва доступні, ці два поняття еквівалентні\parencite{buchholtz:2023}.
\end{rem}

Звісно, ми можемо визначити тотожний морфізм $\ArrId{x} : x \to x$ як $\lambda \_.\,x$. Більше того, кожна
функція $f : X \to Y$ автоматично має дію на синтетичні морфізми $\alpha : \Int \to X$ шляхом
пост-композиції $f \circ \alpha : \Int \to Y$. В такому випадку ми часто пишемо $f\prn{\alpha}$.

З $\Int$ ми негайно отримуємо $n$-куби $\Int^n$ і з них ми можемо виділити симплекси
$\Delta^n$, межі $\partial \Delta^n$ та роги $\Lambda_k^n$. Зокрема, $\Delta^2 \to X$
представляє $2$-комірку в $X$, що свідчить про композицію двох стрілок,
і $\Lambda^2_1 \to X$ представляє пару композиційних стрілок (без композиції). Ми наводимо означення
цих типів нижче:
\begin{mathparpagebreakable}
  \Delta^n = \Compr{\prn{i_1, \dots, i_n} : \Int^n}{i_1 \ge i_2 \ge \dots \ge i_n}
  \and
  \Lambda^2_1 = \Compr{\prn{i,j} : \Int^2}{i = 1 \lor j = 0}
\end{mathparpagebreakable}

\begin{nota}
  Ми пишемо $i : \Delta^n$ ($0 \le i \le n$) як скорочений запис для послідовності з $i$ копій $1$, за якими слідує $0$: $\prn{1, 1, \dots, 0, \dots}$.
\end{nota}

Відображення $f : \Delta^2 \to X$ називається свідченням того, що композиція $f\prn{-,0}$, за якою йде $f\prn{1,-}$, дорівнює $\lambda i.\,f\prn{i,i}$. Ми наголошуємо, що це --- \emph{дані}; може існувати багато різних відображень $f$, що свідчать про одну й ту саму композицію, оскільки $X$ може мати багато нееквівалентних $2$-клітин із тією ж границею. З цієї ж причини, однак, не завжди пара компонованих морфізмів $\Lambda^2_1 \to X$ поширюється до даних композиції $\Delta^2 \to X$. Це пояснюється саме тим, що не кожен тип у \STT{} можна розглядати як категорію; хоча ми визначили $\Hom[X]{x}{y}$ для кожного $X$, не існує апріорного способу \emph{компонування} цих морфізмів. Прекатегорії --- це типи, для яких існують усі композиції:
\begin{defi}
  \emph{Прекатегорія} --- це тип $X$, що задовольняє умову Сегала:
  вкладення $\Lambda^2_1 \to \Delta^2$ індукує еквівалентність $\IsEquiv\prn{X^{\Delta^2} \to X^{\Lambda^2_1}}$.
\end{defi}
Грубо кажучи, умова Сегала гарантує, що кожна пара компонованих морфізмів в $X$ поширюється (унікально) до $2$-клітини, що свідчить про їх композицію, і, зокрема, існує індукована функція композиції $\Hom{x}{y} \times \Hom{y}{z} \to \Hom{x}{z}$. Унікальність автоматично гарантує, що ця операція є асоціативною та унітальною. Визначення категорії уточнює це дещо далі. У прекатегорії $X$ ми можемо визначити тип ізоморфізмів $x \cong y$ між $x,y : X$, і тому існує два потенційно різних типи свідчень того, що $x$ та $y$ є ідентичними: $x =_X y$ та $x \cong_X y$. Категорія --- це прекатегорія, для якої ці два типи є канонічно еквівалентними.
\begin{defi}
  $\alpha : \Hom{x}{y}$ є ізоморфізмом ($\IsIso\prn{\alpha}$), якщо існують $\beta_0,\beta_1 : \Hom{y}{x}$, такі що $\beta_0 \circ f = \ArrId{}$, $f \circ \beta_1 = \ArrId{}$.\footnote{Це саме еквівалентність у сенсі \HoTT{}, але перенесена на синтетичні морфізми.} Ми пишемо $\Iso{x}{y}$ або $x \cong y$ для підтипу ізоморфізмів.
\end{defi}
\begin{defi}
  \label{def:prelims:category}
    Прекатегорія $C$ є \emph{категорією}, якщо вона задовольняє умову Резка:
    \[
      \Prod{\DeclVar{x,y}{}{C}} \IsEquiv\prn{\DeclVar{\mathsf{idtoiso}}{}{\prn{x=y} \to \Iso{x}{y}}}
    \]
    де $\mathsf{idtoiso}(\mathsf{refl}) := \ArrId{}$.
    Якщо кожен морфізм у $C$ є ізоморфізмом, то $C$ є \emph{групоїдом}.
\end{defi}
\begin{exa}
  $\Int,\Delta^n,\Int^n$ є категоріями~\parencite{gratzer:2024}.
\end{exa}

\begin{lem}
  $C$ є групоїдом тоді й лише тоді, коли $\IsEquiv\prn{\DeclVar{C^{!_\Int}}{}{C \to C^\Int}}$
  \textup($C$ є \emph{$\Int$-нульовим} \textup{\cite{rijke:2020})}.
\end{lem}

\textcite{riehl:2017} розвивають базову теорію цих синтетичних категорій. Як зазначено вище, кожна функція має дію на морфізмах, і \opcit{} показує, що ця дія зберігає композиції та тотожні морфізми, а отже, визначає функтор. Вони також показують, що $C \to D$ є категорією, якщо $D$ є категорією, і що синтетичні морфізми $\Hom[D^C]{f}{g}$ є саме природними перетвореннями. Можна переформулювати різні класичні категорні поняття досить безпосередньо:
\begin{defiC}[\parencite{bardomiano:2021}]
  Природне перетворення $\alpha : \Hom[C^I]{\Const\prn{c}}{F}$ свідчить про те, що $c$ є границею для $F : C^I$, якщо $\alpha$ індукує еквівалентність $\Hom{c'}{c} \Equiv \Hom{\Const\prn{c'}}{F}$ для всіх $c'$.
\end{defiC}

\begin{defi}
  Спряженість між двома категоріями $C,D$ складається з пари функцій $f : C \to D$ та $g : D \to C$ разом із природним ізоморфізмом $\iota : \Prod{c,d} \Hom{f\prn{c}}{d} \Equiv \Hom{c}{g\prn{d}}$.
\end{defi}

Хоча вище ми навели кілька прикладів категорій, визначним типом, який \emph{не є} категорією, є універсум $\Uni$. Відображення $A : \Int \to \Uni$ є занадто неструктурованими для компонування і, зокрема, не відповідають ні функціям $A\prn{0} \to A\prn{1}$, ні функціям $A\prn{1} \to A\prn{0}$ (розгляньте $\lambda i.\, i = 0$ або $\lambda i.\, i = 1$). У Розділі~\ref{sec:prelims:modal-stt} ми обговоримо підуніверсум $\Space$, побудований раніше~\parencite{gratzer:2024}, який є категорією групоїдів, чиї морфізми відповідають функціям. Щоб належним чином обґрунтувати це визначення, ми нагадаємо, що означає для $X : A \to \Uni$ бути коваріантним сімейством~\parencite{riehl:2017}, що дає зіставлення морфізмам $\Hom{a_0}{a_1}$ функцій $X\prn{a_0} \to X\prn{a_1}$.

\begin{nota}
  Для сімейства $X : A \to \Uni$ ми пишемо $\TotalType{X}$ для $\Sum{a : A} X\prn{a}$.
\end{nota}

\begin{defi}
  Сімейство $X : A \to \Uni$ є \emph{коваріантним},
  якщо для кожного $a : \Hom{a_0}{a_1}$ та $x_0 : X\prn{a_0}$
  наступний тип є стягуваним:
  \[
    \Con{Lift}\prn{a,x_0} =
    \Sum{x_1 : X\prn{a_1}}
    \Sum{x : \Hom{\prn{a_0,x_0}}{\prn{a_1,x_1}}}
    \Proj[1]\prn{x} =_{\Hom{a_0}{a_1}} a
  \]
  Тут $x$ --- морфізм у $\TotalType{X}$. Надалі вираз $\Sum{a : A} X\prn{a} \to A$ є коваріантною проєкцією, коли $X$ є коваріантним сімейством. Для довільного відображення $\pi : X \to A$ ми пишемо $X_a$ для $\Sum{x : X} \pi\prn{x} = a$ і кажемо, що $\pi$ є коваріантним, коли коваріантним є $\lambda a.\,X_a$.
\end{defi}

Оскільки $\Con{Lift}\prn{a,x_0}$ є стягуваним, він має мешканця $x_1$. Це дає функцію $a,x_0 \mapsto x_1$, яка визначає відображення перенесення $a_! : X\prn{a_0} \to X\prn{a_1}$. Стягуваність $\Con{Lift}\prn{a,x_0}$ гарантує, що ці функції компонуються правильно, \etc{}

\begin{lem}
  \label{lem:prelims:covariant}
  Сімейство $\DeclVar{X}{}{A \to \UU}$ є коваріантним тоді й лише тоді, коли відображення $\bar{\pi} := \lambda p.\,\prn{p\prn{0}, \Proj[1] \circ p} : \TotalType X^\Int \to \TotalType X \times_A A^\Int$ є еквівалентністю.
\end{lem}

У Розділах~\ref{sec:adj:pw} та \ref{sec:kan:quillen-a} ми будемо коротко використовувати ослаблення коваріантності:
\begin{defi}
  Сімейство $X : A \to \Uni$ є кодекартовим, якщо відображення ${\TotalType{X}}^\Int \to A^\Int \times_{A^{\brc{1}}} {\TotalType{X}}^{\brc{1}}$ має праве спряжене $\ell \Adjoint \bar{\pi}$, таке що $\bar{\pi} \circ \ell = \ArrId{}$.
\end{defi}
Можна дати еквівалентне визначення через кодекартові морфізми та показати, наприклад, що кожен морфізм у $D$ може бути розкладений у композицію кодекартового морфізму, за яким йде вертикальний морфізм:
\begin{thm}[\textcite{buchholtz:2023}]
  Якщо відображення $\pi : D \to C$ є кодекартовим, то для кожного $c : \Int \to C$ та $x_0 : C_{c(0)}$ категорія $\Con{Lift}\prn{c,x_0}$ має початковий об'єкт.
\end{thm}
Двоїсто можна розглядати контраваріантні та декартові сімейства та розшарування.

Нарешті, ми зауважимо, що оскільки категорії та групоїди визначаються через певні умови \emph{ортогональності}, за \textcite{rijke:2020} вони визначають рефлексивні підуніверсуми.
\begin{prop}
  Існують ідемпотентні монади $\Rezkify,\Grpdify$, такі що, наприклад, $\Rezkify X$ є категорією і $C^{\Rezkify X} \Equiv C^X$, коли $C$ є категорією.
\end{prop}

\begin{prop}[\textcite{riehl:2017}]
  \label{thm:prelims:stt-models}
  При спеціалізації Теореми~\ref{thm:prelims:hott-models} на симпліціальні простори \textup($\PSH{\SIMP}$\textup), отримана модель задовольняє Аксіому~\ref{ax:int}, і в цій моделі категорії реалізуються $\infty$-категоріями \textup(модельованими повними просторами Сегала\textup), а групоїди --- $\infty$-групоїдами.
\end{prop}


\subsection{Модальна гомотопічна теорія типів}
Багато теорем у теорії категорій вимагають можливості квантифікації за об'єктами категорії, наприклад: «якщо $\alpha : F \to G$ є природним перетворенням функторів $\CC \to \DD$ і кожен $\alpha_c$ є оборотним, то $\alpha$ є оборотним». Версію цього твердження було доведено у \textcite{riehl:2017}: $\prn{\Prod{c : C} \IsIso\prn{\lambda i.\,\alpha\,i\,c}} \to \IsIso\prn{\alpha}$, проте воно є тонко відмінним, як ми обговоримо нижче. Насправді, у теперішньому вигляді ми не можемо безпосередньо виразити класичне твердження в \STT{}.

Щоб зрозуміти розбіжність між результатами в \STT{} та класичними результатами, зауважимо, що працюючи внутрішньо в теорії типів при доведенні $\Prod{c : C} \IsIso\prn{\lambda i.\,\alpha\,i\,c}$, ми не можемо припустити, що $c$ є просто об'єктом у $C$: оскільки це довільний елемент, ми змушені припускати, що він побудований у довільному контексті, який може містити, наприклад, копію $\Int$, так що $c$ представляє синтетичний морфізм. Дійсно, якщо ми розгорнемо вищевказаний тип у моделі, то виявимо, що побудова $\Prod{c : C} \IsIso\prn{\lambda i.\,\alpha\,i\,c}$ вже передбачає доведення того, що обрані обернені є природними.Значна частина сили симпліціальної теорії типів випливає саме з цієї неявної природності, але це робить цей конкретний результат слабшим. Зрештою, його мета у стандартній теорії категорій полягала в тому, що в цій конкретній ситуації \emph{a priori} неприродні вибори обернених автоматично стають природними. Крім того, ми зіткнемося з теоремами, які є просто хибними при такому наївному перекладі.

Відповідно, щоб зробити \STT{} практичною, ми маємо розширити її \emph{модальностями}: унарними конструкторами типів, що вирізняються тим, що вони не зберігають підстановку або не застосовуються в довільних контекстах. Наприклад, згодом ми оснастимо \STT{} модальністю $\Modify[\GM]{-}$, яка відкидає всі необоротні синтетичні морфізми з типу, щоб отримати його \emph{ядро} (core), яке ми потім використаємо для точного кодування поточкової оборотності (див. Приклад~\ref{ex:prelim:pw-naturality}).

Повний опис модальної теорії типів, яку ми використовуємо --- \MTT{}~\parencite{gratzer:2020} --- наведено у \textcite{gratzer:phd}, а формальні правила ми фіксуємо в Розділі~\ref{sec:mtt-rules}. На щастя, додавання модальностей не впливає на правила, наприклад, для $\Sum{}$-типів. Відповідно, для стислості ми нагадаємо лише нові правила, які необхідно додати до \MLTT{}, щоб розширити \HoTT{} модальностями в стилі \MTT{}.

\MTT{} параметризується \emph{теорією мод} (mode theory): строгою $2$-категорією, що описує набір доступних модальностей (морфізмів) разом із природними перетвореннями між ними ($2$-клітинами). Ми використовуємо $\mu,\nu,\xi$ для позначення модальностей. У випадку симпліціальної теорії типів наша теорія мод матиме лише один об'єкт разом із кількома твірними модальностями та $2$-клітинами. Існує чотири твірні модальності $\GM,\SM,\OM,\TM$, що задовольняють такі рівняння:
\begin{mathparpagebreakable}
  \GM = \GM \circ \GM = \GM \circ \SM = \GM \circ \OM = \TM \circ \GM %
  \and
  \SM = \SM \circ \SM = \SM \circ \GM = \SM \circ \OM %
  \and
  \OM \circ \OM = \ArrId{}
\end{mathparpagebreakable}

Ми також вимагаємо наявності таких твірних $2$-клітин:
\begin{mathparpagebreakable}
  \epsilon : \GM \to \ArrId{} \qquad \zeta : \ArrId{} \to \SM
  \and
  \tau : \TM \cong \TM \circ \OM \quad\text{(разом із $\tau^{-1})$}
  \and
  \pi_0^\TM : \TM \to \OM
  \and
  \pi_1^\TM : \TM \to \ArrId{}
\end{mathparpagebreakable}

Ці $2$-клітини так само задовольняють низку рівнянь. Для $\epsilon$ та $\zeta$ ми вимагаємо $\zeta \Whisker \SM = \SM \Whisker \zeta = \ArrId{}$, розглядаючи всі ці перетворення як $2$-клітини $\SM \to \SM$, та $\GM \Whisker \zeta = \ArrId{} : \GM \to \GM$ (де $\Whisker$ позначає віскірінг). Ми вимагаємо двоїстих рівнянь для $\epsilon$. Для решти чотирьох $2$-клітин ми вимагаємо, щоб такі діаграми були комутативними:
\[
  \begin{tikzpicture}[diagram]
    \node (T) {$\TM$};
    \node [below = of T] (OT) {$\TM \circ \OM$};
    \node [left = 2.5cm of OT] (OC) {$\OM$};
    \node [right = 2.5cm of OT] (C) {$\ArrId{}$};
    \path[->] (T) edge node[right] {$\tau$} (OT);
    \path[->] (OT) edge node[below] {$\pi^\TM_0 \Whisker \OM$} (C);
    \path[->] (OT) edge node[below] {$\pi^\TM_1 \Whisker \OM$} (OC);
    \path[->] (T) edge node[above] {$\pi^\TM_1$} (C);
    \path[->] (T) edge node[above] {$\pi^\TM_0$} (OC);

    \node [below = of T] (OT) {$\TM$};
    \node [left = 2.5cm of OT] (OC) {$\OM$};

Кожен морфізм $\mu$ у теорії мод індукує модальний тип $\Modify[\mu]{-}$. Правила для цих модальних типів ми опишемо за мить, але спочатку дамо уявлення про те, що вони покликані позначати. Наразі це лише інтуїція, хоча аксіоми та модель, описані в Розділі~\ref{sec:prelims:modal-stt}, зроблять її строгою. Як уже зазначалося, $\Modify[\GM]{-}$ видаляє всі нетотожні синтетичні морфізми з типу. $\Modify[\SM]{-}$ є правим спряженим до цієї операції, тому він відкидає всі нетотожні морфізми, але потім вільно додає всі морфізми, так що $n$-симплекс $\Delta^n \to \Modify[\SM]{X}$ є саме набором із $n$ точок в $X$.\footnote{Зауважте, що хоча $\Modify[\GM]{X}$ завжди буде $\infty$-категорією (насправді $\infty$-групоїдом), це не так для $\Modify[\SM]{X}$. Зокрема, $\Modify[\SM]{X}$ майже ніколи не задовольнятиме умову Резка.} Далі, $\Modify[\OM]{-}$ переводить тип у його протилежний і, зокрема, змінює напрямки всіх синтетичних морфізмів. Нарешті, $\Modify[\TM]{-}$ переводить тип у відповідний йому тип \emph{скручених стрілок}; ми детальніше проаналізуємо його в Розділі~\ref{sec:tw}.

Правило утворення для $\Modify[\mu]{-}$ є складним: весь сенс модальностей полягає в тому, що з $\Gamma \vdash A$ не випливає $\Gamma \vdash \Modify{A}$. Замість цього \MTT{} вводить новий вид операції над контекстом, яка діє як «лівий спряжений» до $\Modify[\mu]{-}$:
\begin{mathparpagebreakable}
  \inferrule{
    \IsCx{\Gamma}
  }{
    \IsCx{\LockCxV{\Gamma}}
  }
  \and
  \inferrule{
    \IsTy[\LockCxV{\Gamma}]{A}
  }{
    \IsTy{\Modify{A}}
  }
  \and
  \inferrule{
    \IsTm[\LockCxV{\Gamma}]{a}{A}
  }{
    \IsTm{\MkMod{a}}{\Modify{A}}
  }
\end{mathparpagebreakable}
Ми називаємо $\brc{\mu}$ \emph{модальним обмеженням} (modal restriction). Корисно порівняти $\Modify[\mu]{A}$ з залежними добутками й, відповідно, розглядати $\LockCxV{-}$ як розширення контексту чимось схожим на субструктурну «$\mu$-змінну»~\parencite{bahr:2017,birkedal:2020}. Справжня сила модальностей виявляється в тому, як ці $\brc{\mu}$ взаємодіють зі змінними. Зокрема, загалом не виконується $\Gamma, x : A, \brc{\mu} \vdash x : A$; оскільки $\LockCxV{-}$ має моделювати лівий спряжений, ми не можемо в загальному випадку припустити існування підстановки ослаблення $\LockCxV{\Gamma} \to \Gamma$. Замість цього ми змінюємо правило розширення контексту змінною так, що кожна змінна анотується модальністю:
\begin{mathparpagebreakable}
  \inferrule{
    \IsCx{\Gamma}
    \\
    \IsTy[\LockCxV{\Gamma}]{A}
  }{
    \IsCx{\MECxV{\Gamma}{x}{A}}
  }
  \and
  \inferrule{
    \alpha : \mu \to \Con{mods}\prn{\Gamma_1}
  }{
    \IsTm[\Gamma_0, \DeclVar{x}{\mu}{A}, \Gamma_1]{x^\alpha}{A^\alpha}
  }
\end{mathparpagebreakable}
У наведеному вище правилі $x^\alpha$ --- це нова форма правила для змінних, тоді як $A^\alpha$ --- допустима операція над синтаксисом, яка обходить терм $A$ й належним чином оновлює всі вільні змінні $y^\beta$, що зустрічаються в $A$, модифікуючи $\beta$ за допомогою $\alpha$. Зокрема, якщо $A$ є закритим, то $A^\alpha = A$. У формальному синтаксисі як $x^\alpha$, так і $A^\alpha$ реалізуються за допомогою певної форми підстановки, детальніше див.~\textcite{gratzer:mtt-journal:2021}.

Оригінальне розширення контексту виходить при $\mu = \ArrId{}$. У другому правилі $\Con{mods}\prn{\Gamma_1}$ --- це композиція $\nu_0 \circ \nu_1 \circ \cdots$ усіх $\brc{\nu_i}$, що зустрічаються в $\Gamma_1$ (і дорівнює $\ArrId{}$, якщо таких немає). Іншими словами, змінну з анотацією $\mu$ можна використовувати саме тоді, коли вона стоїть за послідовністю модальних обмежень, для яких існує $2$-клітина, що веде від $\mu$ до цієї композиції. Отже, саме в правилі для змінних починають грати роль $2$-клітини.
\begin{lem}
  Якщо $\IsTy[\Gamma, \DeclVar{x}{\mu}{A}]{B}$, $\IsTm[\Gamma, \DeclVar{x}{\mu}{A}]{b}{B}$ та $\IsTm[\LockCxV{\Gamma}]{a}{A}$, то виконується $\IsTy{\Sb{B}{a/x}}$ та $\IsTm{\Sb{b}{a/x}}{\Sb{B}{a/x}}$.
\end{lem}

Останньою частиною мозаїки є правило елімінації для модальностей. Грубо кажучи, це правило стверджує, що модальні анотації еквівалентні модальним типам «з точки зору типу», тобто завдання елемента в контексті $\MECxV{\Gamma}{x}[\nu]{\Modify[\mu]{A}}$ є тим самим, що й завдання елемента в $\MECxV{\Gamma}{x}[\nu\circ\mu]{A}$. Конкретно це зводиться до такого правила зіставлення зі зразком, яке дозволяє нам припустити, що $\DeclVar{x}{\nu}{\Modify[\mu]{A}}$ має вигляд $\MkMod{y}$, де $\DeclVar{y}{\nu\circ\mu}{A}$:
\begin{mathparpagebreakable}
  \inferrule{
    \IsTy[\MECxV{\Gamma}{x}[\nu]{\Modify{A}}]{B}
    \\
    \IsTm[\MECxV{\Gamma}{y}[\nu\circ\mu]{A}]{b}{\Sb{B}{\MkMod[\mu]{y}/x}}
    \\
    \IsTm[\LockCxV{\Gamma}[\nu]]{a}{\Modify{A}}
  }{
    \IsTm{\LetMod{a}[y]{b}}{\Sb{B}{a/x}}
  }
  \and
  \inferrule{
    \IsTy[\MECxV{\Gamma}{x}[\nu]{\Modify{A}}]{B}
    \\
    \IsTm[\MECxV{\Gamma}{y}[\nu\circ\mu]{A}]{b}{\Sb{B}{\MkMod[\mu]{y}/x}}
    \\
    \IsTm[\LockCxV{\Gamma}[\nu\circ\mu]]{a}{A}
  }{
    \EqTm{\prn{\LetMod{\MkMod{a}}[y]{b}}}{\Sb{b}{a/y}}{\Sb{B}{\MkMod{a}/x}}
  }
\end{mathparpagebreakable}

Хоча ці правила покривають усі необхідні розширення для роботи з модальними типами, ми також користуємося додатковою зручною можливістю --- модальними $\Prod{}$-типами:
\begin{mathparpagebreakable}
  \inferrule{
    \IsTm[\MECxV{\Gamma}{x}{A}]{b}{B}
  }{
    \IsTm{\lambda x.b}{\Prod{\DeclVar{x}{\mu}{A}} B}
  }
  \and
  \inferrule{
    \IsTm{f}{\Prod{\DeclVar{x}{\mu}{A}} B}
    \quad
    \IsTm[\LockCxV{\Gamma}]{a}{A}
  }{
    \IsTm{f\prn{a}}{\Sb{B}{a/x}}
  }
\end{mathparpagebreakable}

\begin{nota}
  «Якщо $\DeclVar{c}{\GM}{C}$, то $\Phi\prn{c}$» означає $\Prod{\DeclVar{c}{\GM}{C}} \Phi\prn{c}$.
\end{nota}

\begin{exa}
  \label{ex:prelim:pw-naturality}
  Точний переклад твердження «поточкова оборотність тягне за собою оборотність», де $\DeclVar{C,D}{\GM}{\Uni}$ та $\alpha : C \times \Int \to D$, має вигляд:
  $\prn{\Prod{\DeclVar{c}{\GM}{C}} \IsIso\prn{\lambda i.\,\alpha\,i\,c}} \to \IsIso\prn{\alpha}$
\end{exa}

Безпосередньо з цих правил ми можемо довести таке твердження:
\begin{prop}[\textcite{gratzer:2020}]
  \label{thm:prelims:mtt-facts}
  \leavevmode
  \begin{itemize}
  \item $\Modify[\mu]{-}$ комутує з $\Sum{}$ та $\ObjTerm{}$
  \item $\Con{comp} : \Modify[\mu]{\Modify[\nu]{-}} \Equiv \Modify[\mu\circ\nu]{-}$ та $\Modify[\ArrId{}]{-} \Equiv \ArrId{}$
  \item Якщо $\alpha : \mu \to \nu$, то існує відображення $\Coe^{\alpha} : \Modify[\mu]{{-}} \to \Modify[\nu]{-^\alpha}$.
  \item $\Con{transp} : \Modify[\GM]{\Modify[\GM]{A} \to B} \Equiv \Modify[\GM]{A \to \Modify[\SM]{B}}$
  \item $\Con{transp} : \Modify[\GM]{\Modify[\OM]{A} \to B} \Equiv \Modify[\GM]{A \to \Modify[\OM]{B}}$
  \end{itemize}
  Перший пункт дає функцію $\prn{\circledast} : \Modify{A \to B} \to \Modify{A} \to \Modify{B}$.
\end{prop}

Коли це не викликає плутанини, ми опускатимемо еквівалентності $\Modify[\mu]{\Modify[\nu]{A}} \Equiv \Modify[\mu\circ\nu]{A}$ та $\Modify[\ArrId{}]{A} \Equiv A$. Крім того, оскільки немає двозначності, ми опускаємо $\epsilon$ (та її віскірінги) і просто пишемо $x$ замість $x^\epsilon$, і аналогічно для $\zeta : \ArrId{} \to \SM$. Як наслідок, якщо $\DeclVar{A}{\GM}{\Uni}$, то ми можемо просто писати $\Modify[\TM]{A}$ замість $\Modify[\TM]{A^{\TM\Whisker\epsilon}}$. За домовленістю ми також уникаємо написання $x^{\ArrId{}}$.

\begin{nota}
  Якщо $\IsTm[\LockCxV{\Gamma}]{f}{A \to B}$, ми пишемо $f^\dagger$ для функції $\MkMod[\mu]{f} \mathbin{\circledast} -$.
\end{nota}

\begin{rem}\label{rem:prelims:coe-natural}
  Оскільки ми будемо неодноразово цим користуватися, зауважимо, що $\Coe^\alpha$ завжди є належним чином природним. Наприклад, зафіксуємо $\DeclVar{f}{\GM}{A \to B}$. Тоді ми будуємо шлях $\alpha : \Coe^{\pi_1^\TM} \circ f^\dagger = f \circ \Coe^{\pi_1^\TM}$ таким чином:
  \[
    \alpha = \Con{funext}\prn{\lambda x.\,\LetMod[\TM]{x}[x_0]{\Refl}}
  \]
  Оскільки цей шлях по суті є комутуючим перетворенням (він задається індукцією, щоб дозволити $\Coe^{\pi_1^\TM}$ та $f^\dagger$ редукуватися), він є повністю когерентним, а вищі шляхи аналогічно будуються за індукцією, а потім рефлексивністю.
\end{rem}

У загальному випадку $\Modify[\mu]{-}$ не обов'язково комутує з пропозиційною рівністю. Проте це так у наших запланованих моделях, тому ми вводимо це як аксіому:
\begin{restatable}{axiom}{crispidaxiom}
  Відображення $\MkMod[\mu]{a} = \MkMod[\mu]{b} \to \Modify{a = b}$, що переводить $\Refl$ в $\MkMod{\Refl}$, є еквівалентністю для всіх $\DeclVar{a,b}{\mu}{A}$.%
\end{restatable}

Якщо бути дуже точними, це відображення визначається індукцією шляхів на сімействі типів $\lambda\prn{x,y:\Modify{A}}.\,\LetMod{x}[a]{\LetMod{y}[b]{\Modify{a = b}}}$. Згідно з \textcite{gratzer:normalization:2022}, для цього принципу існує обчислювальне обґрунтування.

\begin{cor}
  Кожен конструктор $\Modify[\mu]{-}$ комутує з волокнистими добутками (pullbacks)
  $A \times_C B = \Sum{a : A} \Sum{b : B} f\prn{a} =_C g\prn{b}$.
\end{cor}

\begin{rem}
  Для читачів, знайомих із \emph{просторовою теорією типів} (spatial type theory)~\parencite{shulman:2018}, ця модальна теорія типів є розширенням просторової теорії типів, що включає дві додаткові модальності ($\OM$, $\TM$). Зокрема, результати \textcite{shulman:2018}, які стосуються $\GM$ та $\SM$, можуть бути відтворені в цьому контексті.
\end{rem}
\subsection{Модальності та симпліціальна теорія типів}
\label{sec:prelims:modal-stt}

Щоб пов'язати модальну та симпліціальну структури, ми вводимо такі аксіоми, мотивовані запланованою моделлю, як описано в Теоремі~\ref{thm:prelims:stt-models} (і більш загально $\mathcal E^{\Op{\SIMP}}$ для $\infty$-топосу $\mathcal E$); див. також роботу \textcite{myers:2023}. По-перше, протилежне відображення має бути антиеквівалентністю для $\Int$:

\begin{restatable}{axiom}{intopaxiom}
  \label{ax:op-of-int-is-int}
  Існує еквівалентність $\neg : \Modify[\OM]{\Int} \to \Int$, яка міняє місцями $0$ та $1$, а також $\lor$ та $\land$.
\end{restatable}
\begin{cor}
  Ми можемо розширити $\neg$ до еквівалентності $\neg : \Modify[\OM]{\Delta^n} \Equiv \Delta^n$.
\end{cor}

Далі, ми вимагаємо, щоб два можливі поняття дискретності (бути $\Int$-нульовим або $\GM$-модальним) збігалися:

\begin{restatable}{axiom}{intdiscaxiom}
  \label{ax:discrete-iff-crisp}
  Якщо $\DeclVar{A}{\GM}{\Uni}$, то $\Modify[\GM]{A} \to A$ є еквівалентністю \textup($A$ є \emph{дискретним}\textup) тоді й лише тоді, коли $A \to A^\Int$ є еквівалентністю \textup($A$ є \emph{$\Int$-нульовим}\textup).
\end{restatable}
\begin{restatable}{axiom}{intglobalaxiom}
  \label{ax:global-points}
  Канонічне відображення $\Bool \to \Int$ є ін'єктивним та індукує еквівалентність $\Bool \Equiv \Modify[\GM]{\Int}$.
\end{restatable}
Мотивовані нашим запланованим класом моделей, ми наполягаємо на тому, що еквівалентності спільно детектуються за допомогою $\Delta^n$:

\begin{restatable}{axiom}{cubessepaxiom}
  \label{ax:cubes-separate}
  $\DeclVar{f}{\GM}{A \to B}$ є еквівалентністю тоді й лише тоді, коли виконується:
    \[
      \Prod{\DeclVar{n}{\GM}{\Nat}}
      \IsEquiv\prn{
        \prn{f_*}^\dagger : \Modify[\GM]{\Delta^n \to A}
        \to
        \Modify[\GM]{\Delta^n \to B}
      }
    \]
\end{restatable}
\noindent
Зауважимо, що оскільки існує пара переріз-ретракція $\Delta^n \to \Int^n \to \Delta^n$, ми можемо замінити $\Delta^n$ на $\Int^n$ у наведеному вище принципі.

Одним із корисних застосувань є таке твердження:
\begin{lem}
  \label{lem:prelims:covariance-simplices}
  Відображення $\DeclVar{\pi}{\GM}{X \to A}$ є коваріантним тоді й лише тоді, коли відображення
  $\Modify[\GM]{X^{\Delta^n}} \to \Modify[\GM]X \times_{\Modify[\GM]A} \Modify[\GM]{A^{\Delta^n}}$,
  індуковане $\prn{0^*}^\dagger$ та $\prn{\pi_*}^\dagger$, є еквівалентністю для всіх $\DeclVar n\GM\Nat$.
\end{lem}


\subsubsection*{Аксіома для $\Modify[\TM]{-}$}
Нарешті, ми додаємо до \STT{} нову аксіому, яка керує модальністю $\TM$. Для мотивації ми нагадаємо деякі факти про зовнішнє визначення функтора скручених стрілок (twisted arrow functor) $\PSH{\SIMP} \to \PSH{\SIMP}$, який $\Modify[\TM]{-}$ покликаний інтерналізувати. Класично $\Con{Tw} : \PSH{\SIMP} \to \PSH{\SIMP}$ визначається так:
\[
  \Con{Tw}\prn{X}\prn{\brk{n}} = X\prn{\Op{\brk{n}} \ast \brk{n}}
\]
Тут ми написали $\Op{\brk{n}} \ast \brk{n}$ замість еквівалентного $\brk{2n + 1}$, щоб прояснити дію цього функтора на морфізмах: $f \mapsto \Op{f} \ast f$. (Тобто це відповідає операції з'єднання --- join --- на скінченних лінійних порядках.)

Оскільки цей функтор визначений через передкомпозицію, він є правим спряженим, чий лівий спряжений визначається за допомогою лівого розширення Кана. Зокрема, він переводить $\Delta^n : \PSH{\SIMP}$ в $\Delta^{2n + 1}$ (знову ж таки, з функторіальною дією, заданою скручуванням). Таким чином, існує універсальне відображення $\Mor[\eta_n]{\Delta^n}{\Con{Tw}\prn{\Delta^{2n+1}}}$, яке при розгортанні задається тотожним відображенням $\Mor{\brk{2n + 1}}{\brk{2n + 1}}$. Універсальність відображення $\eta_n$ зводиться до вимоги, щоб кожен морфізм $\Mor[f]{\Delta^n}{\Con{Tw}\prn{C}}$ розкладався як $\Con{Tw}\prn{\hat{f}} \circ \eta_n$ для деякого єдиного $\Mor[\hat{f}]{\Delta^{2n + 1}}{C}$. Наша аксіома, що керує $\Modify[\TM]{-}$, аксіоматизує це відображення $\eta_n$ разом із тією властивістю, що $\Con{Tw}\prn{-} \circ \eta_n : \Hom{\Delta^{2n+1}}{C} \to \Hom{\Delta^n}{\Con{Tw}\prn{C}}$ є еквівалентністю. Маючи під рукою цю зовнішню мотивацію, ми переходимо до фіксації деяких позначень та формулювання аксіоми, яка керує модальністю $\Modify[\TM]{-}$.

\begin{nota}
  Якщо $\DeclVar{n}{\GM}{\Nat}$, ми маємо канонічні відображення $i_l : \Delta^n \to \Delta^{2n + 1}$,
  $i_r : \Delta^n \to \Delta^{2n + 1}$ та $i_m : \Delta^1 \to \Delta^{2n + 1}$, які виділяють
  підмножини $\brc{0, \dots, n}$, $\brc{n + 1, \dots, 2n + 1}$ та $\brc{n, n + 1}$ відповідно. Для
  зручності ми пишемо
  $\bar{i}_l = i_l^\dagger \circ \neg : \Delta^n \to \Modify[\OM]{\Delta^{2n + 1}}$.
\end{nota}

Нарешті, у формулюванні нової аксіоми нам потрібна процедура, яка розширює відображення $f : {\Delta^n \to \Delta^m}$ до відображення ${\Delta^{2n + 1} \to \Delta^{2m + 1}}$, що діє належним чином на образах двох вкладень $i_l,i_r : \Delta^n \to \Delta^{2n + 1}$. Щоб обґрунтувати це формально, ми вводимо \emph{притуплене з'єднання} (blunt join) $X \BluntJoin Y$:
\[
  X \BluntJoin Y = X \Pushout{X \times \brc{0} \times Y} \prn{X \times \Int \times Y} \Pushout{X \times \brc{1} \times Y} Y
\]
Це орієнтована версія з'єднання $X \star Y$~\parencite[Розд. 6]{hottbook}, така що $X \BluntJoin Y$ є приблизно $X \Coprod{} Y$ із доданими морфізмами, які з'єднують кожен $x:X$ з кожним $y:Y$.
\begin{lem}
  \label{lem:prelims:blunt-join-of-simplices}
  Якщо $C$ є категорією, то
  $C^{\Delta^{m+1+n}} \Equiv C^{\Delta^{m} \BluntJoin \Delta^n}$.
\end{lem}
\begin{defi}
  Якщо $\DeclVar{f}{\GM}{\Delta^n \to \Delta^m}$, ми визначаємо
  $\Con{twist}\prn{f} : {\Delta^{2n + 1} \to \Delta^{2m + 1}}$ як відображення, отримане шляхом єдиного
  продовження відображення
  $f^\dagger \BluntJoin f : \Modify[\OM]{\Delta^n} \BluntJoin {\Delta^n} \to \Modify[\OM]{\Delta^m}
  \BluntJoin {\Delta^m}$ уздовж категорних еквівалентностей
  $\Modify[\OM]{\Delta^i} \BluntJoin {\Delta^i} \to \Delta^{2i + 1}$.
\end{defi}

\begin{restatable}{axiom}{twarraxiom}
  \label{ax:twisted-arrow-equiv}
  Для кожного $\DeclVar{n}{\GM}{\Nat}$ існує (обов'язково єдина) функція
  $\DeclVar{\eta_n}{\GM}{\Delta^n \to \Modify[\TM]{\Delta^{2n+1}}}$, така що наступне відображення є
  еквівалентністю для кожної категорії $\DeclVar{C}{\GM}{\Uni}$:
  \[
    \iota \defeq \lambda \MkMod[\GM]{f}.\,\MkMod[\GM]{f^\dagger \circ \eta_n}
    :
    \Modify[\GM]{\Delta^{2n + 1} \to C}
    \to
    \Modify[\GM]{\Delta^{n} \to \Modify[\TM]{C}}
  \]
  Крім того, ми вимагаємо, щоб
  $\tau = \prn{\Coe^{\neg}}^\dagger : \Modify[\TM]{\Delta^{n}} \to \Modify[\TM]{\Modify[\OM]{\Delta^{n}}}$ і
  щоб діаграми на Рисунку~\ref{fig:prelims:tm-diagrams} були комутативними \textup(це просто властивості --- всі
  об'єкти є множинами, оскільки $\Modify[\mu]{-}$ зберігає h-рівень\textup).
\end{restatable}

За цієї відповідності $\eta$ є єдиним відображенням $\Delta^n \to \Modify[\TM]{\Delta^{2n + 1}}$, заданим тотожним відображенням $\ArrId{} : \Delta^{2n + 1} \to \Delta^{2n + 1}$, і, таким чином, є універсальним $n$-симплексом. Відображення $\pi^\TM_1$ виділяє нижній рядок, а $\pi^\TM_0$ виділяє верхній рядок, проте \emph{скручений} (twisted) так, що він потрапляє в $\Modify[\OM]{C}$ замість $C$. Ця аксіома використовуватиметься лише при доведенні Теореми~\ref{thm:tw:tw}, де ми використовуємо $\Modify[\TM]{-}$ для побудови біфункторіальної версії $\hom$.

\begin{prop}[\textcite{gratzer:2024}]
  Модель, побудована в Теоремі~\ref{thm:prelims:stt-models}, розширюється до моделі модальної \HoTT{}, яка задовольняє наші аксіоми.
\end{prop}
\begin{rem}
  Хоча наша попередня робота~\parencite{gratzer:2024} не розглядала модальність $\Modify[\TM]{-}$, методи, використані там, безпосередньо масштабуються на цю ситуацію. Зокрема, \textcite{mukherjee:2023} дають явний опис необхідної операції скручених стрілок і показують, що вона є правим спряженим Квіллена, що й потрібно для розширення моделі.
\end{rem}

Маючи під рукою модальності, можна безпосередньо довести низку результатів класичної теорії категорій. Наприклад, так звану основну теорему теорії $\infty$-категорій:

\begin{restatable}{thm}{ffesssurj}
  \label{thm:prelims:ff-ess-surj-to-equiv}
  Якщо $\DeclVar{C,D}{\GM}{\Uni}$ є категоріями, то $\DeclVar{F}{\GM}{C \to D}$ є еквівалентністю, якщо (1) індуковане відображення $\Modify[\GM]{C} \to \Modify[\GM]{D}$ є сюр'єктивним, та (2) для будь-яких $\DeclVar{c,c'}{\GM}{C}$ відображення $\Hom{c}{c'} \to \Hom{F\prn{c}}{F\prn{c'}}$ є еквівалентністю.
\end{restatable}
\begin{proof}
  Припустимо, що умови (1) та (2) виконуються. Ми доведемо, що $F$ є еквівалентністю, використовуючи Аксіому~\ref{ax:cubes-separate}, для чого зафіксуємо $\DeclVar{n}{\GM}{\Nat}$ і покажемо, що відображення
  $F_*^\dagger : \Modify[\GM]{\Delta^n \to C} \to \Modify[\GM]{\Delta^n \to D}$ є еквівалентністю.

  Якщо $n = 0$, то за умовою (1) відображення $F_*^\dagger$ є сюр'єктивним, а за умовою (2) у поєднанні з умовою Резка воно є вкладенням. Відповідно, $F_*^\dagger$ є еквівалентністю в цьому випадку. Випадок $n = 1$ є безпосереднім наслідком випадків для $n = 0$ разом з умовою (2). Загалом, оскільки $C^{\Delta^n} \Equiv C^{\Delta^1} \times_{C} \dots \times_C C^{\Delta^1}$ за умовою Сегала (і аналогічно для $D$), а $\Modify[\GM]{-}$ комутує з волокнистими добутками, випадок для $n \ge 2$ випливає з випадків $n=0,1$.
\end{proof}

\subsection{Базові блоки для побудови категорій}
Finally, we recall two results from our earlier work~\parencite{gratzer:2024} that will be used repeatedly within this work
to construct new categories. The first is a construction of \emph{full subcategories} using $\SM$:

\begin{prop}\label{thm:prelims:full-subcat}
  %
  If $\DeclVar{C}{\GM}{\Uni}$ is a category and $\DeclVar{\phi}{\GM}{\Modify[\GM]{C} \to \Prop}$ is
  a predicate, then 
  \begin{enumerate}
  \item $C_\phi = \Sum{c : C} \Modify[\SM]{\phi\prn{\MkMod[\GM]{c}}}$ is a category,

  \item the projection map $C_\phi \to C$ induces an equivalence on hom-types,

  \item $\Modify[\GM]{C_\phi} \Equiv \Sum{c : \Modify[\GM]{C}} \phi\prn{c}$, and 

  \item a map $\DeclVar{F}{\GM}{D \to C}$ factors through $C_\phi$ if and only if
    $\phi\prn{\MkMod[\GM]{F\prn d}}$ holds for all $\DeclVar{d}{\GM}{D}$.
\end{enumerate}
\end{prop}

\begin{cor}\label{cor:prelims:crisp-ff-is-ff}
  %
  If $\DeclVar{C,D}{\GM}{\Uni}$ are categories and $\DeclVar{F}{\GM}{C \to D}$, then the canonical
  map $\Hom{c}{c'} \to \Hom{F\prn{c}}{F\prn{c'}}$ is an equivalence for all $c,c' : C$ (notice the
  lack of $\GM$!) if and only if it is an equivalence when $\DeclVar{c,c'}{\GM}{C}$.
  %
\end{cor}

Next, we recall the construction of the category of groupoids which plays the role of the
category of sets in simplicial type theory, \eg{}, we shall use this category to define presheaves:

\begin{prop}
  \label{thm:prelims:space} There is a category $\DeclVar{\Space_i}{\GM}{\Uni[i + 1]}$ with an
  embedding $\Space_i \to \Uni[i]$ such that:
  \begin{itemize}
    \item If $X : A \to \Space_i$, then the composite $A \to \Uni[i]$ is covariant.
    \item The converse holds for $\DeclVar{A}{\GM}{\Uni[i]}$, $\DeclVar{X}{\GM}{A \to \Uni[i]}$: if
      $X$ is covariant, then $X$ factors through $\Space_i$.
  \end{itemize}
\end{prop}

\begin{cor}[Directed univalence]
  \label{cor:prelims:dua}
  $\Space^\Int \Equiv \Sum{X_0,X_1 : \Space} X_1^{X_0}$ and composition in $\Space$ is the
  composition of functions.
\end{cor}

\begin{cor}
  If $\DeclVar{X}{\GM}{\Uni}$ is a groupoid, then $X : \Space$.
\end{cor}

\begin{rem}
  We proved~\parencite{gratzer:2024} Proposition~\ref{thm:prelims:space} in a richer variation of \STT{}
  (\emph{triangulated type theory}). Since we only require the result here, we take it as an ``axiom''
  of sorts to work in a simpler type theory and note that one could extend \STT{} to triangulated
  type theory to prove this theorem outright.
\end{rem}

\section{Вкладення Йонеди}
\label{sec:tw}

У цьому розділі ми фіксуємо категорію $\DeclVar{C}{\GM}{\Uni}$. Нашою метою є вивчення типу $\PSH{C} = \Space^{\Modify[\OM]{C}}$ передпучків на $C$. Оскільки $\Space$ є категорією, $\PSH{C}$ також є категорією, і за орієнтованою унівалентністю:
\begin{lem}
  \label{lem:tw:hom-of-psh}
  Якщо $F,G : \PSH{C}$, то
  $\Hom{F}{G} \Equiv \Prod{\DeclVar{c}{}{\Modify[\OM]{C}}}{F\,c \to G\,c}$
\end{lem}

\begin{rem}
  Так само, як, наприклад, із повнотою, $\PSH{C}$ неявно фіксує рівень універсуму так, що $\PSH{C} = \Modify[\OM]{C} \to \Space_i$. Ми можемо розглядати $i$ як параметр або просто покласти $i = 0$. Іноді нам доведеться вимагати, щоб $C \Equiv C'$, де $C' : \Uni[i]$, і в таких ситуаціях ми будемо говорити, що категорія $C$ є \emph{малою}. Ми припускаємо, що всі категорії є локально малими --- тобто кожен тип $\Hom[C]{c}{c'}$ є малим.
\end{rem}

Можна переформулювати \emph{розшаровану} (fibrational) лему Йонеди, доведену у \textcite{riehl:2017}, щоб скористатися перевагами $\PSH{C}$ замість квантифікації за контраваріантними сімействами, як у \opcit:
\begin{lem}
  \label{lem:tw:fibrational-yoneda}
  Якщо $F : \PSH{C}$ та $c : \Modify[\OM]{C}$, то
  $F\prn{c} \cong \Prod{c' : \Modify[\OM]{C}} \Hom[\Modify[\OM]{C}]{c}{c'} \to F\prn{c'}$
\end{lem}

\subsection{Категорія скручених стрілок та вкладення Йонеди}

У світлі цього останнього результату природним наступним кроком є визначення відображення $C \to \PSH{C}$, яке переводить $c : C$ у щось на зразок $\Hom{-}{c}$.\footnote{Тут ми бачимо, чому $C$ має бути плоським (flat): ми хочемо обговорювати як $C$, так і $\Modify[\OM]{C}$. Корисно розуміти $\DeclVar{C}{\GM}{\Uni}$ як \emph{закритий} тип, який не залежить ні від чого в контексті й, зокрема, не потребує функторіального розгляду.} Однак потрібна обережність: $\Hom{-}{c}$ має тип $C \to \Uni$, а не необхідний $\Modify[\OM]{C} \to \Space$. Поміркувавши, читач може виявити дивним те, що відображення $\Hom{-}{-} : C \times C \to \Uni$ взагалі існує; якщо всі відображення є функторіальними в \STT{}, як $\Hom{-}{-}$ може бути коваріантним за обома аргументами? Насправді це є наслідком дивної поведінки синтетичних морфізмів у $\Uni$. Хоча $\Hom{-}{-}$ є функторіальним за обома аргументами, відсутність орієнтованої унівалентності для $\Uni$ робить це марним. Ця дивність гарантує, що $\Hom{-}{-}$ не звужується до функції зі значеннями в $\Space$.

Натомість потрібна функція $\Phi : \Modify[\OM]{C} \times C \to \Space$, така що $\Phi\prn{\MkMod[\OM]{c},-} = \Hom{c}{-}$ завжди, коли $\DeclVar{c}{\GM}{C}$, тобто функція, яка збігається на об'єктах із $\Hom{-}{-}$ і має таку саму функторіальність за другим аргументом, але приймає $\Modify[\OM]{C}$ як свій перший аргумент. Насправді зовсім не очевидно, звідки така функція має взятися; \textcite[с.~xii]{riehl:2022} спеціально виділяють цю конструкцію як надзвичайно тонку в теорії $\infty$-категорій. Саме з цієї причини ми ввели модальність $\Modify[\TM]{-}$. Згадаймо візуалізацію для $\Modify[\GM]{\Delta^n \to \Modify[\TM]{C}}$:
\begin{equation}
  \label{eq:tw:visual}
  \begin{tikzpicture}[diagram,baseline=(link)]
    \node (A0) {$c_n$};
    \node [right = of A0] (A1) {$c_{n-1}$};
    \node [right = of A1] (A2) {$\cdots$};
    \node [right = of A2] (A3) {$c_0$};
    \node [below = 1cm of A0] (B0) {$c_{n+1}$};
    \node [right = of B0] (B1) {$c_{n+2}$};
    \node [right = of B1] (B2) {$\cdots$};
    \node [right = of B2] (B3) {$c_{2n}$};
    \path[<-] (A0) edge (A1);
    \path[<-] (A1) edge (A2);
    \path[<-] (A2) edge (A3);
    \path[->] (B0) edge (B1);
    \path[->] (B1) edge (B2);
    \path[->] (B2) edge (B3);
    \path[->] (A0) edge coordinate[pos=.5] (link) (B0);
  \end{tikzpicture}
\end{equation}
Проєкція на $\Modify[\OM]{C}$ дає верхній рядок, а відображення в $C$ дає нижній. Ця візуалізація для $n$-симплексів є дуже схожою на візуалізацію для $C^{\Int} = \Sum{c_0,c_1} \Hom{c_0}{c_1}$, проте верхній рядок було скручено, щоб гарантувати, що одне обмеження потрапляє в $\Modify[\OM]{C}$, як це необхідно для біфункторіальної версії $\Hom{-}{-}$:

\begin{restatable}{thm}{twlfib}
  \label{thm:tw:tw}
  Якщо $\DeclVar{C}{\GM}{\Uni}$ є категорією, то виконується таке:
  \begin{itemize}
  \item Відображення
    $\gl{\Coe^{\pi_0^\TM},\Coe^{\pi_1^\TM}} : \Modify[\TM]{C} \to \Modify[\OM]{C} \times C$
    випрямляється (straightens) до $\Phi : \Modify[\TM]{C} \times C \to \Space$.
  \item Для кожного $\DeclVar{c}{\GM}{C}$ відображення
    $\alpha_c : \Hom{\Hom{c}{-}}{\Phi\prn{\MkMod[\OM]{c},-}}$, індуковане лемою Йонеди
    (Лема~\ref{lem:tw:fibrational-yoneda}),
    застосованою до $\iota\prn{\MkMod[\GM]{\ArrId{c}}} : \Phi\prn{\MkMod[\OM]{c},c}$, є
    еквівалентністю.
  \end{itemize}
\end{restatable}

\begin{lem}
  \label{lem:tw:simple-coherence}
  Нехай задано $\DeclVar{f}{\GM}{\Delta^{1} \to C}$ і нехай
  $\bar{f} = \iota\prn{\MkMod[\GM]{f}} : \Modify[\TM]{C}$, тоді існують шляхи:
  \begin{gather*}
    \theta\prn{f}_0 :
    \prn{\Coe^{\pi_0^{\TM}} \circ \Con{extract}\prn{\bar{f}}}\prn{\ast}
    =
    \MkMod[\OM]{f\prn{0}}
    \quad
    \theta\prn{f}_1 :
    \prn{\Coe^{\pi_1^{\TM}} \circ \Con{extract}\prn{\bar{f}}}\prn{\ast}
    =
    f\prn{1}
  \end{gather*}.
  Ці шляхи є природними за $C$, так що, наприклад, два шляхи наступного вигляду, індуковані
  $\theta\prn{g \circ f}_1$ та $\theta\prn{f}_1$ разом із природністю $\Coe^{\pi_1^\TM}$ у $g$, узгоджуються:
  \[
    \prn{\Coe^{\pi_1^{\TM}} \circ \Con{extract}\prn{\iota\prn{\MkMod[\GM]{g \circ f}}}}\prn{\ast}
    =
    g\prn{f\prn{1}}
  \]
  Тут $\ast : \Delta^0$ --- єдиний елемент одиничного типу.
\end{lem}
\begin{proof}
  Ми покажемо другий шлях, оскільки вони є симетричними. Ми визначаємо $\theta_1$, використовуючи природність
  $\Coe^{\pi_1^{\TM}}$ та поведінку $\Coe^{\pi_1^{\TM}}$ на відображенні $\eta$
  з Рисунка~\ref{fig:prelims:tm-diagrams}:
  \begin{align*}
    &\prn{\Coe^{\pi_1^{\TM}} \circ \Con{extract}\prn{\iota\prn{\MkMod[\GM]{f}}}}\prn{\ast}
    \\
    &= \prn{\Coe^{\pi_1^{\TM}} \circ f^\dagger \circ \eta_0}\prn{\ast}
    \\
    &= \prn{f \circ \Coe^{\pi_1^{\TM}} \circ \eta_0}\prn{\ast} && \text{За природністю, Зауваження \ref{rem:prelims:coe-natural}}
    \\
    &= f\prn{1} && \text{За першою діаграмою на Рисунку~\ref{fig:prelims:tm-diagrams}}
  \end{align*}

  Щоб довести, що $\theta_1$ є природним за $C$, ми зауважимо, що термі збігаються з точністю до комутуючого
  перетворення (commuting conversion) правил елімінації для модальних типів. Відповідно, ми можемо довести, що ці два шляхи
  узгоджуються за допомогою індукції за $\eta_0\prn{\ast}$, а потім рефлексивності.
\end{proof}

\begin{proof}[Доведення Теореми~\ref{thm:tw:tw}]
  Ми починаємо з доведення того, що відображення
  $\gl{\Coe^{\pi^\TM_0},\Coe^{\pi^\TM_1}} : \Modify[\TM]{C} \to \Modify[\OM]{C} \times C$ є
  коваріантним сімейством. За Лемою~\ref{lem:prelims:covariance-simplices} наступне відображення, індуковане
  $\brc{0} : \Delta^0 \to \Delta^n$, є еквівалентністю:
  \[
    \epsilon : \Modify[\GM]{\Modify[\TM]{C}^{\Delta^n}} \to
    \prn{
      \DelimMin{1}
      \Modify[\GM]{\Modify[\TM]{C}}
      \times_{\Modify[\GM]{\Modify[\OM]{C}} \times \Modify[\GM]{C}}
      \prn{\Modify[\GM]{\Modify[\OM]{C}^{\Delta^n}} \times \Modify[\GM]{C^{\Delta^n}}}
    }
  \]
  Для зручності ми почнемо із застосування кількох модальних перетворень (зокрема, використовуючи
  $\Con{transp} : \Modify[\GM]{A \to \Modify[\OM]{B}} \Equiv \Modify[\GM]{\Modify[\OM]{A} \to B}$)
  так, що достатньо показати, що таке відображення є еквівалентністю:
  \[
    \epsilon' :
    \Modify[\GM]{\Modify[\TM]{C}^{\Delta^n}} \to
    \prn{
      \DelimMin{1}
      \Modify[\GM]{\Modify[\TM]{C}}
      \times_{\Modify[\GM]{C} \times \Modify[\GM]{C}}
      \prn{\Modify[\GM]{C^{\Modify[\OM]{\Delta^n}}} \times \Modify[\GM]{C^{\Delta^n}}}
    }
  \]

  Щоб довести це, ми побудуємо комутативну діаграму:
  \begin{equation}
    \DiagramSquare{
      width = 7cm,
      sw = \Modify[\GM]{\Modify[\TM]{C}^{\Delta^n}},
      nw = \Modify[\GM]{C^{\Delta^{2n+1}}},
      se =
      \Modify[\GM]{\Modify[\TM]{C}}
      \times_{\Modify[\GM]{C} \times \Modify[\GM]{C}}
      \prn{\Modify[\GM]{C^{\Modify[\OM]{\Delta^n}}} \times \Modify[\GM]{C^{\Delta^n}}},
      ne =
      \Modify[\GM]{\Delta^1 \to C}
      \times_{\Modify[\GM]{C} \times \Modify[\GM]{C}}
      \prn{\Modify[\GM]{C^{\Modify[\OM]{\Delta^n}}} \times \Modify[\GM]{C^{\Delta^n}}},
      north = \phi,
      south = \epsilon',
      west = \iota,
      east = \prn{\iota,\ArrId{}}
    }
    \label{diag:key-diagram}
  \end{equation}
  Ми визначимо $\phi$ за мить, але спочатку зауважимо, що пара $\prn{\iota,\ArrId{}}$ є
  коректно визначеною завдяки Лемі~\ref{lem:tw:simple-coherence}, яка гарантує, що застосування $\iota$, а потім
  обчислення належним чином комутує з проєкцією.

  Зауважимо також, що два вертикальні відображення є еквівалентностями, оскільки $\iota$ є
  еквівалентністю. Відповідно, за правилом 3-із-2 (3-for-2), щоб показати, що $\epsilon'$ є еквівалентністю,
  достатньо переконатися, що $\phi$ є еквівалентністю, яка робить діаграму комутативною.\footnote{%
    Ми підкреслюємо, що заповнення (filler) для цього квадрата не має значення. Будь-яке заповнення
    є достатнім, щоб показати, що $\epsilon'$ є еквівалентністю, що, у свою чергу, означає, що $\epsilon$ є
    еквівалентністю, як і вимагалося.}
  Тепер ми визначимо $\phi$ так:
  \[
    \phi\prn{\MkMod[\GM]{f}} \defeq
    \prn{
      \MkMod[\GM]{\Restrict{f}{n \le n + 1}},
      \prn{
        \MkMod[\GM]{\Restrict{f}{0 \le \dots \le n} \circ \neg},
        \MkMod[\GM]{\Restrict{f}{n + 1 \le \dots \le 2n + 1}}
      },
      \Refl
    }
  \]
  $\phi$ задається звуженням уздовж категорної еквівалентності
  ($\Modify[\OM]{\Delta^{n}} \BluntJoin \Delta^n \to \Delta^{2n + 1}$),
  тому воно є еквівалентністю.

  Далі зауважимо, що всі чотири відображення на цій діаграмі є слабко природними за $C$. Для нижнього
  та верхнього відображень це просте спостереження --- верхнє задається звуженням, а нижнє використовує
  звуження разом із $\Coe^{\pi_1^{\TM}}$, що також є природним
  за Зауваженням~\ref{rem:prelims:coe-natural}. Для лівого відображення це є наслідком природності
  $\iota$. Для правого відображення єдиною складністю є шляхи, які свідчать про те, що відображення $\iota$
  комутує з обчисленням на проєкціях. Це вимагає заповнення для певного шляху, але це є
  саме тією природною когерентністю, що забезпечується другою частиною Леми~\ref{lem:tw:simple-coherence}.

  Зрештою, ми доводимо, що діаграма комутує. Зафіксуємо
  $\MkMod[\GM]{f} : \Modify[\GM]{\Delta^{2n + 1} \to C}$. Ми хочемо показати, що
  $\epsilon'\prn{\iota\prn{\MkMod[\GM]{f}}}
  = \prn{\iota,\ArrId{}}\prn{\phi\prn{\MkMod[\GM]{f}}}$. Однак, за природністю, ми
  можемо звести задачу за допомогою $f$ до випадку, коли $C = \Delta^{2n + 1}$ та $f = \ArrId{}$. У цьому випадку
  все, що бере участь, є множинами, тому достатньо довести комутативність діаграми, замінивши кожен
  волокнистий добуток (pullback) на простий добуток. З урахуванням цього ми обчислюємо:
  \begin{align*}
    \epsilon'\prn{\iota\prn{\MkMod[\GM]{\ArrId{}}}}
    &= \epsilon'\prn{\MkMod[\GM]{\eta_n}}
    \\
    &=
      \prn{
        \MkMod[\GM]{\eta_n\,0},
        \prn{
          \Con{transp}\prn{\MkMod[\GM]{\Coe^{\pi_0^{\TM}}\circ \eta_n}},
          \MkMod[\GM]{\Coe^{\pi_1^{\TM}} \circ \eta_n}
        }
      }
    \\
    &=
      \prn{
        \MkMod[\GM]{\eta_n\,0},
        \prn{
          \Con{transp}\prn{\MkMod[\GM]{\bar{i}_l}},
          \MkMod[\GM]{i_r}
        }
      }
    \\
    &=
      \prn{
        \MkMod[\GM]{i_m\prn{\eta_0\,\ast}},
        \prn{
          \MkMod[\GM]{i_l \circ \neg},
          \MkMod[\GM]{i_r}
        }
      }
    \\
    &=
      \prn{\iota,\ArrId{}}\prn{
        \MkMod[\GM]{i_m},
        \prn{
          \MkMod[\GM]{i_l \circ \neg},
          \MkMod[\GM]{i_r}
        }
      }
    \\
    &= \prn{\iota,\ArrId{}}\prn{\phi\prn{\MkMod[\GM]{\ArrId{}}}}
  \end{align*}

  Це завершує перший крок доведення. Другий крок полягає в тому, щоб показати, що для кожного
  $\DeclVar{c}{\GM}{C}$ відображення
  $\alpha_c : \Hom[C \to \Space]{\Hom{c}{-}}{\Phi\prn{\MkMod[\OM]{c},-}}$ є ізоморфізмом. Переходячи
  до тотальних просторів, достатньо показати, що наступне відображення є еквівалентністю:
  \[
    \tilde{\alpha}_c = \lambda \prn{d,f}.\,\prn{d,f_*\prn{\iota\prn{\MkMod[\GM]{\ArrId{c}}}}}
    : \Sum{d : C} \Hom{c}{d} \to \Sum{d : C} \Phi\prn{\MkMod[\OM]{c},d}
  \]
  У наведеному вище виразі $f_*$ --- це операція коваріантного переносу (transport) на $\Phi\prn{\MkMod[\OM]{c},-}$.
  Оскільки обидві сторони цього відображення є категоріями, достатньо показати, що це відображення є
  строго повним (fully faithful) та істотно сюр'єктивним (essentially surjective).

  Дійсно, $\tilde\alpha_c$ є еквівалентністю на об'єктах. Для цього зауважимо, що якщо
  $\DeclVar{f}{\GM}{\Hom{c}{d}}$ для деякого
  $\DeclVar{d}{\GM}{C}$, то ми можемо побудувати перенос $f_*$ альтернативно наступним чином.
  Визначимо шлях $h : \Delta^1 \to \Modify[\TM]{C}$ як $h \defeq\iota{\prn{\MkMod[\GM]{\lambda \_, \_, k.\, f\prn{k}}}}$,
  тобто
  $h$ відповідає такому подвійно виродженому (degenerate) 3-симплексу в $C$:
  \[
    \begin{tikzcd}
      c\ar[d,"\ArrId{c}"'] & c\ar[l,"\ArrId{c}"'] \\
      c\ar[r,"f"'] & d
    \end{tikzcd}
  \]
  Потім ми розглядаємо морфізм $\lambda i.\,\prn{\pi^\TM_1\prn{h\,i}, h\,i}$ в
  $\Sum{d : C} \Phi\prn{\MkMod[\OM]{c},d}$. Використовуючи визначення $\iota$ та природність
  $\eta$, це відображення є морфізмом від $\prn{c,\iota\prn{\MkMod[\GM]{\ArrId{c}}}}$ до
  $\prn{d, \iota\prn{\MkMod[\GM]{f}}}$. Крім того, природність відображення $\Coe^{\pi_1^{\TM}}$ гарантує,
  що воно лежить над $f$ у $C$. Як наслідок,
  $\tilde{\alpha}_c\prn{d,f} = \prn{d,\iota\prn{\MkMod[\GM]{f}}}$ при обмеженні на
  $\DeclVar{d}{\GM}{C}$ та $\DeclVar{f}{\GM}{\Hom{c}{d}}$, що є еквівалентністю, оскільки $\iota$
  є оборотним.

  Для повної унівалентності достатньо показати, що таке відображення є еквівалентністю:
  \[
    \tilde{\alpha}_c :
    \Modify[\GM]{\Int \to \Sum{d : C} \Hom{c}{d}}
    \to
    \Modify[\GM]{\Int \to \Sum{d : C} \Phi\prn{\MkMod[\OM]{c},d}}
  \]
  Однак, оскільки обидві сторони є тотальними просторами коваріантних сімейств, достатньо показати, що
  наступне відображення є еквівалентністю:
  \[
    \tilde{\alpha}_c :
    \Modify[\GM]{\Sum{d : \Int \to C} \Hom{c}{d\,0}}
    \to
    \Modify[\GM]{\Sum{d : \Int \to C} \Phi\prn{\MkMod[\OM]{c},d\,0}}
  \]
  Тепер висновок випливає з попереднього випадку.
\end{proof}

\begin{nota}
  Ми пишемо $\Phi_D : \Modify[\OM]{D} \times D \to \Space$ для тієї ж конструкції, застосованої до деякої категорії $D$. У межах цього розділу ми продовжуємо писати $\Phi$ як скорочення для $\Phi_C$.
\end{nota}

\begin{cor}
  \label{cor:tw:tw-of-op}
  Якщо $\DeclVar{c_0}{}{\Modify[\OM]{C}}$ та $c_1 : C$, то
  $\Phi\prn{c_0,c_1} = \Phi_{\Modify[\OM]{C}}\prn{\MkMod[\OM]{\MkMod[\OM]{c_1}},c_0}$.
\end{cor}
\begin{proof}
  Переходячи до тотальних просторів, достатньо знайти еквівалентність $\Modify[\TM]{C} \to
  \Modify[\TM]{\Modify[\OM]{C}}$, яка вписується в таку діаграму:
  \[
    \begin{tikzpicture}[diagram]
      \node (T) {$\Modify[\TM]{C}$};
      \node[right = 5cm of T] (OT) {$\Modify[\TM]{\Modify[\OM]{C}}$};
      \node[below right = 1.75cm and 2.5cm of T] (C) {$\Modify[\OM]{C} \times C$};
      \path[->] (T) edge (C);
      \path[->] (OT) edge (C);
      \path[->] (T) edge (OT);
    \end{tikzpicture}
  \]

  Відображення $\Coe^\tau$ точно задовольняє цю роль: воно є оборотним, оскільки $2$-клітина $\tau$ є ізоморфізмом, і воно вписується в комутативну діаграму через відповідну діаграму в теорії мод.
\end{proof}

\subsection{Лема Йонеди}

Маючи під рукою біфункторіальну версію $\Hom{-}{-}$, ми тепер можемо безпосередньо визначити вкладення Йонеди $\Yo$ та використати Лему~\ref{lem:tw:fibrational-yoneda} для отримання результату про $\Yo$:
\begin{defi}[Йонеда]
   $\Yo = \lambda c.\,\Phi\prn{-,c} : C \to \PSH{C}$.
\end{defi}
\begin{lem}
  \label{lem:tw:weak-yoneda}
  $\Hom{\Yo{c}}{X} \cong X\prn{\MkMod[\OM]{c}}$ для всіх $X : \PSH{C}$ та $\DeclVar{c}{\GM}{C}$.
\end{lem}
\begin{proof}
    Оскільки $c$ анотовано модальністю $\GM$, використовуючи Теорему~\ref{thm:tw:tw} та Наслідок~\ref{cor:tw:tw-of-op}, ми маємо таку ідентифікацію: $\Hom[\Modify[\OM]{C}]{\MkMod[\OM]{c}}{-} = \Phi\prn{-,c}$. Крім того, за Лемою~\ref{lem:tw:hom-of-psh} ми додатково маємо:
    \[
      \Prod{c' : \Modify[\OM]{C}} \Phi\prn{c',c} \to X\prn{c'} \cong \Hom{\Yo{c}}{X}
    \]
    Тепер висновок випливає з Леми~\ref{lem:tw:fibrational-yoneda}.
\end{proof}

Значна частина теорії категорій міститься в Лемі~\ref{lem:tw:weak-yoneda}. Вона показує, що $\Yo$ є строго повним на $\GM$-анотованих елементах $C$ і що $C$ є повною підкатегорією в $\PSH{C}$:
\begin{lem}
  \label{lem:tw:repr}
  $\Yo : C \to \PSH{C}$ індукує еквівалентність $C \Equiv \PSH{C}_{\Con{isRepr}}$, де
  $\Con{isRepr} = \lambda X.\,\Sum{c : C} X = \Yo{c}$.\footnote{Зауважте, що $\Con{isRepr}\prn{X}$ є судженням (proposition) внаслідок Леми~\ref{lem:tw:weak-yoneda} та Наслідку~\ref{cor:prelims:crisp-ff-is-ff}.}
\end{lem}

Хоча Лема~\ref{lem:tw:weak-yoneda} безпосередньо випливає з Леми~\ref{lem:tw:fibrational-yoneda}, вищевказаний наслідок можна сформулювати лише тоді, коли існує \emph{категорія} передпучків --- те, що було відсутнє у \textcite{riehl:2017}. Це відкриває нову стратегію доведення: щоб довести твердження для $C$, ми спочатку доводимо, що воно виконується для $\Space$, потім для $\PSH{C}$, а потім показуємо, що воно звужується на повну підкатегорію. Наприклад, ми можемо довести згадану раніше характеристику природних ізоморфізмів:
\begin{thm}
  \label{thm:tw:nat-iso}
  Якщо $\DeclVar{C,D}{\GM}{\Uni}$ є категоріями, $\DeclVar{F,G}{\GM}{C \to D}$ та $\DeclVar{\alpha}{\GM}{\Hom{F}{G}}$, то $\Prod{c : C} \IsIso\prn{\alpha\,c}$ виконується, якщо виконується $\Prod{\DeclVar{c}{\GM}{C}} \IsIso\prn{\alpha\,c}$.
\end{thm}
\begin{proof}
  Зауважимо, що ця теорема є тривіальною для $C = \Delta^0$, а для $C = \Delta^1, D = \Space$ вона є наслідком Наслідку~\ref{cor:prelims:dua}. Умова Сегала для $\Space$ тоді тягне за собою теорему для $C = \Delta^n, D = \Space$.

  За Лемою~\ref{lem:tw:repr} достатньо припустити, що $D = \PSH{D_0}$. За Аксіомою~\ref{ax:discrete-iff-crisp} та Теоремою~\ref{thm:prelims:mtt-facts} достатньо показати, що $\prn{\Sum{c : C} \IsIso\prn{\alpha\,c}} \to \prn{\Sum{c : C} \Modify[\SM]{\IsIso\prn{\alpha\,c}}}$ є еквівалентністю. За Аксіомою~\ref{ax:cubes-separate} достатньо довести для всіх $n$, що наступне відображення є еквівалентністю:
  \[
    \IsEquiv\Bigl(\Modify[\GM]{\prn{\Sum{c : C} \IsIso\prn{\alpha\,c}}^{\Delta^n}} \to \Modify[\GM]{\prn{\Sum{c : C} \Modify[\SM]{\IsIso\prn{\alpha\,c}}}^{\Delta^n}}\Bigr)
  \]
  Розгортаючи та комутуючи $\GM$ із $\Sum{}$, достатньо показати, що для кожного $\DeclVar{c}{\GM}{\Delta^n \to C}$ виконується:
    \[
      \Sum{\DeclVar{\sigma}{}{\Delta^n}} \IsIso\prn{\alpha\prn{c\,\sigma}}
      \Equiv
      \Sum{\DeclVar{\sigma}{\GM}{\Delta^n}} \IsIso\prn{\alpha\prn{c\,\sigma}}
    \]
  Проте заміна $\alpha$ на $\alpha \circ c$ зводить нас до вже доведеного випадку $C = \Delta^n, D = \Space$.
\end{proof}

Лема~\ref{lem:tw:weak-yoneda} вже є потужною. Проте вона не фіксує того, що ця еквівалентність є \emph{природною} за обома аргументами $c$ та $X$ --- або, точніше, оскільки $c$ є $\GM$-анотованим, а еквівалентність знаходиться в $\Uni$, природність, яку вона дає, є тривіальною. Ми можемо довести набагато сильнішу версію леми Йонеди, яка (1) не вимагає припущення, що $\DeclVar{c}{\GM}{C}$, і (2) дає бажану функторіальність як за $c$, так і за $X$. Для цього ми замінюємо $\Hom{-}{-}$ на $\Phi$:

\begin{restatable}[Функторіальна лема Йонеди]{thm}{functorialyoneda}
  \label{thm:tw:strong-yoneda}
  Існує природний ізоморфізм
  $\Phi_{\PSH{C}}\prn{\Yo^\dagger\prn{-}, -} \cong \Con{eval} : \Modify[\OM]{C} \times \PSH{C} \to \Space$.
\end{restatable}
\begin{rem}
  Цей результат використовує кілька результатів із Розділу~\ref{sec:adj}. Ці посилання вперед є виправданими: ми не використовуємо Теорему~\ref{thm:tw:strong-yoneda} до Розділу~\ref{sec:adj:psh}. Ми наводимо доведення тут задля концептуальної цілісності.
\end{rem}
\begin{proof}
  Основна складність у цьому доведенні полягає в тому, щоб знайти відображення
  $\Phi_{\PSH{C}}\prn{\Yo^\dagger\prn{-}, -} \to \Con{eval}$, для якого потім можна перевірити, що воно є
  еквівалентністю. Для побудови цього відображення ми використовуємо представлення
  $\Modify[\OM]{C} \times \PSH{C} \to \Space$ як коваріантних сімейств над
  $\Modify[\OM]{C} \times \PSH{C}$. Зокрема, ми розглядаємо такі діаграми волокнистих добутків (pullback):
  \[
    \begin{tikzpicture}[diagram]
      \node (PhiPsh) {$\Modify[\TM]{\PSH{C}}$};
      \node [below = 2cm of PhiPsh] (PshPsh) {$\Modify[\OM]{\PSH{C}} \times \PSH{C}$};
      \node [pullback,left = 3.6cm of PhiPsh] (V) {$V$};
      \node [left = 3.6cm of PshPsh] (CPsh) {$\Modify[\OM]{C} \times \PSH{C}$};
      \node [pullback,left = 3.6cm of V] (PhiC) {$\tilde{\Phi}_C$};
      \node [left = 3.6cm of CPsh] (CC) {$\Modify[\OM]{C} \times C$};
      \path[->] (PhiPsh) edge (PshPsh);
      \path[->] (V) edge (PhiPsh);
      \path[->] (V) edge (CPsh);
      \path[->] (PhiC) edge node[above] {$v$} (V);
      \path[->] (PhiC) edge (CC);
      \path[->] (CPsh) edge node[below] {$\Yo^\dagger \times \ArrId{}$} (PshPsh);
      \path[->] (CC) edge node[below] {$\ArrId{} \times \Yo$} (CPsh);
    \end{tikzpicture}
  \]
  \[
    \begin{tikzpicture}[diagram]
      \node (Spt) {$\Sum{A : \Space} A$};
      \node [below = 2cm of Spt] (S) {$\Space$};
      \node [pullback,left = 3.75cm of Spt] (V) {$W$};
      \node [left = 3.75cm of S] (CPsh) {$\Modify[\OM]{C} \times \PSH{C}$};
      \node [pullback,left = 3.75cm of V] (PhiC) {$\Modify[\TM]{C}$};
      \node [left = 3.75cm of CPsh] (CC) {$\Modify[\OM]{C} \times C$};
      \path[->] (Spt) edge (S);
      \path[->] (V) edge (Spt);
      \path[->] (V) edge (CPsh);
      \path[->] (PhiC) edge node[above] {$w$} (V);
      \path[->] (PhiC) edge (CC);
      \path[->] (CPsh) edge node[below] {$\Con{eval}$} (S);
      \path[->] (CC) edge node[below] {$\ArrId{} \times \Yo$} (CPsh);
    \end{tikzpicture}
  \]
  Тоді стверджується, що $V \Equiv W$. Щоб показати це, ми доводимо, що якщо ми замінимо композицію
  $\Modify[\TM]{C} \to \Modify[\OM]{C} \times C \to \Modify[\OM]{C} \times \PSH{C}$ вільним
  коваріантним сімейством, то відображення $\bar{v},\bar{w}$, індуковані $v$ та $w$, обидва будуть еквівалентностями.
  Тоді висновок показує, що $\bar{v} \circ \bar{w}^{-1}$ є шуканою еквівалентністю.

  За відомим результатом \textcite[Теорема 2.41]{rijke:2020}, існує вільне коваріантне розшарування
  (free covariant fibration) $Z : \Modify[\OM]{C} \times \PSH{C} \to \Uni$, і якщо $\DeclVar{c}{\GM}{C}$ та
  $\DeclVar{X}{\GM}{\PSH{C}}$, то за Наслідком~\ref{cor:adj:push-pull-formula} ми маємо:
  \begin{align*}
    &Z\prn{c, X} =
    \\
    &\ \Grpdify\prn{\Sum{c_0,c_1} \Hom{c_0}{c} \times \Hom{\Yo{c_1}}{X} \times \Phi\prn{c_0,c_1}}
  \end{align*}
  Щоб показати, наприклад, що $v$ індукує еквівалентність, ми маємо показати, що таке відображення є
  еквівалентністю:
  \[
    Z\prn{c,X} \to \Hom{\Yo^\dagger\prn{c}}{X}
  \]
  Ми можемо скористатися Теоремою~\ref{thm:tw:nat-iso} та припустити, що існує $\DeclVar{c'}{\GM}{C}$, такий що
  $c = \MkMod[\OM]{c'}$, та що $\DeclVar{X}{\GM}{\PSH{C}}$. Крім того, оскільки права частина є
  групоїдом, це відображення однозначно індукується продовженням канонічного відображення такого типу:
  \begin{align*}
    &\prn{\Sum{c_0,c_1} \Hom{c_0}{c} \times \Hom{\Yo{c_1}}{X} \times \Phi\prn{c_0,c_1}}
    \\
    &\ \to \Phi\prn{\Yo^\dagger\prn{c},X} \Equiv X\prn{c'}
  \end{align*}
  Це відображення переводить $\prn{c_0,c_1,f,\alpha,t}$ в $\alpha\,c\,\prn{\Phi\prn{f,\ArrId{}}\,t}$, і можна
  безпосередньо перевірити, що присвоєння $x \mapsto \eta\prn{c,c',\ArrId{},\ArrId{},F_x}$ є
  квазіоберненим до цього відображення, де $F_x : \Hom{\Yo\prn{c'}}{X}$ відповідає
  $x : X\prn{\MkMod[\OM]{c'}}$ за Лемою~\ref{lem:tw:weak-yoneda}. Випадок для $w$ аналогічний.
\end{proof}
\section{Ще раз про спряженість}
\label{sec:adj}

Маючи в розпорядженні передпучки та вкладення Йонеди, ми знову звертаємося до теорії спряжених функторів, введеної у \textcite{riehl:2017} в \STT{}. Вони визначають пару функцій $f : C \to D$ та $g : D \to C$ як спряжені, якщо вони забезпечені еквівалентністю $\iota : \Prod{c,d} \Hom{f\prn{c}}{d} \Equiv \Hom{c}{g\prn{d}}$. Хоча вони пропонують кілька еквівалентних переформулювань з використанням природних перетворень одиниці та коодиниці, жодних нетривіальних прикладів спряженостей не наводиться --- що не дивно, оскільки конкретні приклади категорій в \STT{} з'явилися порівняно недавно. Навіть маючи в розпорядженні $\Space$, досить складно побудувати приклади таких спряженостей.

Набагато реалістичніше побудувати лише відображення $f$, а потім показати, що передпучок $\Phi\prn{f^\dagger\prn{-},d} : \PSH{C}$ є представним для кожного $\DeclVar{d}{\GM}{D}$. Це можна порівняти з Теоремою~\ref{thm:tw:nat-iso}: ми хочемо дати функторіальне визначення одного з відображень $f$ або $g$ і \emph{нефункторіальне} визначення іншого, а потім показати, що це відображення можна покращити до повноцінної спряженості. У цьому розділі ми показуємо, що це дійсно можливо, і зауважуємо, що низка важливих спряженостей та результатів тоді стають безпосередньо досяжними. Зокрема, ми використаємо цю техніку, щоб довести, що категорія $\PSH{C}$ є коповною і, більше того, є вільним копоповненням категорії $C$.

\subsection{Поточкові спряженості та спряженості}
\label{sec:adj:pw}

Почнемо з формалізації поняття поточково спряжених (pointwise adjoints):
\begin{defi}
  Ми говоримо, що відображення $\DeclVar{f}{\GM}{C \to D}$ є поточково лівим спряженим, якщо наступний тип є заселеним (inhabited):
  \[
    \Prod{\DeclVar{d}{\GM}{D}} \Con{isRepr}\prn{\Phi\prn{f^\dagger\prn{-}, d}}
  \]
  Дуально, $f$ є поточково правим спряженим, якщо $f^\dagger : \Modify[\OM]{C} \to \Modify[\OM]{D}$ є поточково лівим спряженим.
\end{defi}

Наша основна теорема спирається на два важливі попередні результати. Перший показує, що будь-який поточково лівий спряжений $f$ породжує відображення в іншому напрямку, яке вибирає різні (обов'язково єдині) представні об'єкти для передпучка $\Phi\prn{f^\dagger\prn{-}, d}$.
\begin{lem}
  Якщо $\DeclVar{f}{\GM}{C \to D}$ є поточково лівим спряженим, то тип морфізмів $\DeclVar{g}{\GM}{D \to C}$, забезпечених природним ізоморфізмом $\iota : \Phi\prn{f^\dagger\prn{-}, -} \cong \Yo \circ g$, є стягуваним (contractible).
\end{lem}
\begin{proof}
  Оскільки $\Yo$ є вкладенням, цей тип є судженням (proposition). Тому достатньо показати, що він є заселеним. За припущенням, $\bar{g} = \Phi\prn{f^\dagger\prn{-}, d}$ є представним для всіх $\DeclVar{d}{\GM}{D}$, а отже, пропускається через $\PSH{C}_{\Con{isRepr}}$. Посткомпозиція з еквівалентністю $\PSH{C}_{\Con{isRepr}} \Equiv C$ дає бажане відображення $g : D \to C$.
\end{proof}

Використовуючи це, ми доводимо універсальний випадок теореми, яка покращує поточково спряжений функтор до спряженого: кожне відображення $\DeclVar{g}{\GM}{D \to C}$, що є декартовим розшаруванням~\parencite{buchholtz:2023}, таке що шар (fiber) над кожним $\DeclVar{c}{\GM}{C}$ має початковий (initial) об'єкт~\parencite{bardomiano:2021}, допускає лівий спряжений.

\begin{lem}
  \label{lem:adj:functoriality-of-init-objects}
  %
  Якщо відображення $\DeclVar{g}{\GM}{D \to C}$ є декартовим і для кожного $\DeclVar{c}{\GM}{C}$ шар $D_c$ має початковий об'єкт, то існує відображення $f : C \to D$, таке що $f\prn{c}$ є початковим у $D_c$ для всіх $c : C$.
\end{lem}

\begin{proof}
  Зауважимо, що $\Con{hasInitialObj}\prn{D_c}$ є судженням, і тому за Аксіомою~\ref{ax:discrete-iff-crisp} ми можемо припустити, що $\Modify[\GM]{\Con{hasInitialObj}\prn{D_c}}$ виконується для кожного $\DeclVar{c}{\GM}{C}$. Маючи це спостереження під рукою, ми можемо показати, що $g$ є поточково правим спряженим: якщо $\DeclVar{c}{\GM}{C}$ та $\DeclVar{d}{}{D}$, то:
  \begin{align*}
    \Phi_C\prn{\MkMod[\OM]{c},g\prn{d}}
    &\Equiv \Hom{c}{g\prn{d}}
    \\
    &\Equiv \Hom{\ObjInit{D_c}}{d} && \text{$g$ є декартовим}
    \\
    &\Equiv \Phi_D\prn{\MkMod[\OM]{\ObjInit{D_c}},d}
  \end{align*}

  На цьому останньому кроці ми використовуємо наше спостереження, що виконується $\Modify[\GM]{\Con{hasInitialObj}\prn{D_c}}$, і, що критично важливо, не лише те, що $\Con{hasInitialObj}\prn{D_c}$ виконується. Зокрема, ми спираємося на той факт, що $\DeclVar{\ObjInit{D_c}}{\GM}{D_c}$.

  Відповідно, ми отримуємо функцію $\DeclVar{f}{\GM}{C \to D}$, яка переводить $\DeclVar{c}{\GM}{C}$ в $\ObjInit{D_c}$. Залишається показати, що $f\prn{c}$ є початковим у $D_c$ для всіх $c : C$. Оскільки $D = \Sum{c:C}D_c$, це рівносильно тому, що наступне відображення є еквівалентністю: $\prn{\Sum{d : D} \Hom[D_{g\prn{d}}]{f\prn{g\prn{d}}}{d}} \to D$.

  Щоб довести це, ми використовуємо Теорему~\ref{thm:prelims:ff-ess-surj-to-equiv}, яка дозволяє нам звести задачу до $\GM$-анотованого випадку, де висновок випливає з того факту, що $f\prn{c}$ є тоді початковим у $D_c$.
\end{proof}

\begin{thm}
  \label{thm:adj:pw-to-full-adj}
  Поточково праві спряжені відображення є правими спряженими.
\end{thm}
\begin{proof}
  Для заданого відображення $\DeclVar{g}{\GM}{D \to C}$ розглянемо декартове сімейство
    \[
      \pi : \COMMA{C}{g} = \prn{\Sum{c : C}\Sum{d : D} \Hom{c}{g\prn{d}}} \to C
    \]
  Оскільки $g$ є поточково правим спряженим, кожен шар проекції $\pi$ над $\DeclVar{c}{\GM}{C}$ має початковий об'єкт. Тоді ми застосовуємо Лему~\ref{lem:adj:functoriality-of-init-objects}, щоб отримати відображення $\bar{f} : C \to \COMMA{C}{g}$. Зрештою, композиція $\Proj[2] \circ \bar{f}$ є шуканим лівим спряженим до $g$:
  \begin{align*}
    &Hom[C]{c}{g\prn{d}}
    \\
    &\quad\Equiv \Sum{\alpha : \Hom[C]{c}{g\prn{d}}} \Hom[\COMMA{C}{g}_c]{\bar{f}\prn{c}}{\prn{c, d, \alpha}}
    \\
    &\quad\Equiv
      \Sum{\alpha : \Hom[C]{c}{g\prn{d}}} \Sum{\beta : \Hom[D]{f\prn{c}}{d}}
      g\prn{\beta} \circ \Proj[3]\prn{\bar{f}\prn{c}} = \alpha
    \\
    &\quad\Equiv \Hom[D]{f\prn{c}}{d}
  \end{align*}
  Перший крок використовує початковість $\bar{f}\prn{c}$ у шарі над $c$, а другий розгортає визначення морфізму в $\COMMA{C}{g}$.
\end{proof}

\subsection{Приклади спряженостей}

Ми скористаємося перевагами Теореми~\ref{thm:adj:pw-to-full-adj}, щоб отримати важливі приклади спряжених відображень.
Найбільш значущим є наступний результат:

\begin{thm}
  \label{thm:adj:precomp-is-right-adj}
  Якщо $\DeclVar{f}{\GM}{D \to C}$ і категорія $D$ є малою, то
  $\PSH{f} \defeq (f^\dagger)^* : \PSH{C} \to \PSH{D}$ є правим спряженим із лівим спряженим $f_!$.
\end{thm}
\begin{proof}
  Для простоти позначень ми замінимо $C$ та $D$ на $\Modify[\OM]{C}$ та $\Modify[\OM]{D}$. За Теоремою~\ref{thm:adj:pw-to-full-adj} достатньо припустити, що $\DeclVar{X}{\GM}{D \to \Space}$, і побудувати відображення $f_!\prn{X} : C \to \Space$ разом із природною бієкцією

$\Prod{Y} \Hom{f_!\prn{X}}{Y} \Equiv \Hom{X}{f^*\prn{Y}}$. Це є безпосереднім наслідком \textcite[Теорема 2.41]{rijke:2020} після локалізації композиції $\Sum{c : C} X\prn{C} \to D$ відносно відображення $\brc{0} \to \Int$.
\end{proof}

\begin{cor}
  Лівий спряжений $f_! \Adjoint \PSH{f}$ задовольняє умову $f_! \circ \Yo \cong \Yo \circ f$.
\end{cor}

\begin{cor}
  Простір $\Space$ є мало-коповним: відображення $\Const : \Space \to \Space^C$ є правим спряженим із лівим спряженим $\Colim{}$ для малих категорій $\DeclVar{C}{\GM}{\Uni}$. Явно, якщо $\DeclVar{X}{\GM}{C \to \Space}$, то $\Colim{C} X = \Grpdify\prn{\Sum{c : C} X\prn{c}}$.
\end{cor}
\begin{rem}
  Можна довести, що $\Space$ є повним (тобто $\Const \Adjoint \Lim{}$) за результатом \textcite{gratzer:2024}. Зокрема, вони показують, що $\Prod{c : C} X\prn{c} : \Space$ завжди, коли $X : C \to \Space$ та $\DeclVar{C}{\GM}{\Uni}$. Наслідок~\ref{cor:prelims:dua} та Лема~\ref{lem:tw:hom-of-psh} тоді тягнуть за собою те, що $\Hom{A}{\Prod{c : C} X\prn{c}} \Equiv \Hom{\Con{const}\,A}{X}$.
\end{rem}

\begin{lem}
  \label{lem:adj:eval-at-pt-is-radj}
  Якщо $\DeclVar{c}{\GM}{C}$, то відображення $\PSH{c} : \PSH{C} \to \Space$ є лівим спряженим.
\end{lem}
\begin{proof}
  Для простоти ми знову замінимо $C$ на $\Modify[\OM]{C}$ та застосуємо Теорему~\ref{thm:adj:pw-to-full-adj}. Тоді ми припустимо, що нам задано $\DeclVar{X}{\GM}{\Space}$, і визначимо $c_*X$ як $\lambda c'.\,X^{\Hom{c'}{c}}$. Це відображення є коваріантним за $c'$ \parencite{gratzer:2024}. Тоді достатньо показати, що відображення $X^{\Hom{c}{c}} \to X$, задане обчисленням на тотожному морфізмі, індукує еквівалентність $\Hom[C \to \Space]{Y}{c_*X} \Equiv \Hom[\Space]{Y\prn{c}}{X}$. Це, у свою чергу, є наслідком того факту, що відображення $Y\prn{c} \to \Grpdify\prn{\Sum{c' : C} \Hom{c'}{c} \times Y\prn{c'}}$ є еквівалентністю, тому ми можемо розкласти відображення оцінки (evaluation map) наступним чином:
  \begin{align*}
    &\Hom[C \to \Space]{Y}{c_*X}
    \\
    &\Equiv \Prod{c' : C} Y\prn{c'} \to X^{\Hom{c'}{c}}
    \\
    &\Equiv \Prod{c' : C} \Hom{c'}{c} \times Y\prn{c'} \to X
    \\
    &\Equiv \Grpdify \prn{\Sum{c' : C} \Hom{c'}{c} \times Y\prn{c'}} \to X
    \\
    &\Equiv Y\prn{c} \to X \qedhere
  \end{align*}
\end{proof}

Поєднуючи Теорему~\ref{thm:adj:precomp-is-right-adj} та Лему~\ref{lem:adj:eval-at-pt-is-radj}, ми отримуємо наступне описання для $f_!X$:

\begin{cor}
  \label{cor:adj:push-pull-formula}
  %
  Якщо $\DeclVar{X}{\GM}{\PSH{C}}$, $\DeclVar{f}{\GM}{C \to D}$ та $\DeclVar{d}{\GM}{D}$, то ми можемо явно ототожнити $\prn{f_!X}\,d$ із таким типом:
  \[
    \Grpdify \prn{\Sum{c : \Modify[\OM]{C}} X\prn{c} \times \Hom{f^\dagger c}{d}}
  \]
\end{cor}
\begin{proof}
  Спочатку зауважимо, що $\PSH{d}f_!\prn{X} = \prn{f_!X}\,d$. Транспонуючи, маємо $\Hom{\PSH{d}f_!X}{Z} \Equiv \Hom{X}{\PSH{f}d_*Z}$ для кожного $\DeclVar{Z}{\GM}{\Space}$. Ми обчислюємо $\Hom{X}{\PSH{f}d_*Z}$, використовуючи визначення $d_*$, наведене вище:
  \begin{align*}
    &\Hom[\PSH{C}]{X}{f^*d_*Z}
    \\
    &\Equiv \Prod{\DeclVar{c}{}{\Modify[\OM]{C}}} X\prn{c} \to \Hom{f^\dagger c}{d} \to Z
    \\
    &\Equiv \prn{\Sum{\DeclVar{c}{}{\Modify[\OM]{C}}} X\prn{c} \times \Hom{f^\dagger c}{d}} \to Z
  \end{align*}
  Отже, $\PSH{d}f_!X$ задовольняє універсальну властивість типу

 $\Grpdify \prn{\Sum{\DeclVar{c}{}{\Modify[\OM]{C}}} X\prn{c} \times \Hom{f^\dagger c}{d}}$.
\end{proof}

Наступна лема не вимагає застосування Теореми~\ref{thm:adj:pw-to-full-adj}, а є просто наслідком маніпулювання природними перетвореннями:
\begin{lem}
  \label{lem:adj:comp}
  Якщо відображення $f : C \to D$ є спряженим, то таким є і $f_* : C^A \to D^A$.
\end{lem}

\begin{cor}
  \label{cor:adj:limits-are-pointwise}
  Якщо категорія $C$ є (ко)повною, то такою є і $C^D$, і (ко)границі обчислюються поточково. Зокрема, категорія передпучків $\PSH{C}$ є (ко)повною.
\end{cor}
\begin{cor}
  \label{cor:adj:yo-preserves-limits}
  Вкладення Йонеди зберігає всі границі.
\end{cor}
\begin{proof}
  Якщо $\DeclVar{F}{\GM}{I \to C}$ і границя $\Lim{} F$ існує, то функторіальність $\Yo$ індукує відображення $\Yo{\Lim{} F} \to \Lim{}\prn{\Yo \circ F}$, тож достатньо перевірити, що це відображення є оборотним на всіх $\DeclVar{c}{\GM}{C}$. Розгортаючи, ми маємо довести, що $\Hom{c}{\Lim{} F} \Equiv \Lim{} \Hom{c}{F}$ є еквівалентністю, але це безпосередньо випливає з Леми~\ref{lem:tw:hom-of-psh}.
\end{proof}

Зрештою, ми показуємо, що повні підкатегорії $\Space_{\le n}$ простору $\Space$, визначені $n$-усіченими (truncated) типами, утворюють рефлексивні підкатегорії в $\Space$. Ідея проста: використовувати HIT-типи усічення (truncation HITs). Проте не є автоматичним те, що вони звужуються до відображення $\Trunc{-}_{n} : \Space \to \Space_{\le n}$. Ми доводимо це разом із рефлексивністю $\Space_{\le n}$, використовуючи Теорему~\ref{thm:adj:pw-to-full-adj}:
\begin{cor}
  Вкладення $\Space_{\le n} \to \Space$ є правим спряженим.
\end{cor}
\begin{cor}
  Простір $\Space_{\le n}$ є (ко)повним.
\end{cor}
Така ж методологія застосовується до підкатегорії модальних типів, пов'язаних з ідемпотентною монадою~\parencite{rijke:2020}.

\begin{exa}[Спряження Ісбелла]
  Якщо $\DeclVar{C}{\GM}{\Uni}$, то відображення спряження Ісбелла (Isbell conjugation) $\phi$ є лівим спряженим:
  \begin{align*}
    &\phi : \PSH{C} \to \Modify[\OM]{C \to \Space}
    \\
    &\phi\prn{X} = \MkMod[\OM]{\lambda c.\,\Phi\prn{X,\Yo{c}}}
  \end{align*}
\end{exa}

\subsection{Універсальна властивість категорій передпучків}
\label{sec:adj:psh}

Далі ми узагальнюємо Теорему~\ref{thm:adj:precomp-is-right-adj}, щоб показати, що якщо $\DeclVar{f}{\GM}{C \to E}$, де $C$ --- мала категорія, а $E$ --- коповна категорія, то відображення $\Phi\prn{f^\dagger\prn{-},-} : E \to \PSH{C}$ є правим спряженим, загалом слідуючи аргументам, наведеним у \textcite{cisinski:2019}. Почнемо з кількох загальних лем. Далі ми фіксуємо $C$ та $E$, як зазначено вище.

По-перше, як наслідок доведення Теореми~\ref{thm:adj:precomp-is-right-adj}:
\begin{lem}
  Кограниця вкладення $\Yo : C \to \PSH{C}$ дорівнює $\ObjTerm{\PSH{C}} = \lambda \_.\,\ObjTerm{}$.
\end{lem}

З наведеного вище, а також за подальшого аналізу кограниць, ми можемо вивести результат, що становить незалежний інтерес: кожен передпучок є кограницею представних передпучків.
\begin{lem}[Щільність $\Yo$]
  \label{lem:adj:density}
  Якщо $\DeclVar{X}{\GM}{\PSH{C}}$, то
  $X \cong \Colim{\Modify[\OM]{\TotalType{X}}} \Yo \circ \pi^\dagger$,
  де $\TotalType X=\Sum{c:\Modify[\OM]C}X\prn c$.
\end{lem}
\begin{proof}
  Ми починаємо з наступного обчислення, де $\pi : \TotalType{X} \to \Modify[\OM]{C}$ та $\pi^\dagger_! : \Space^{\TotalType{X}} \to \PSH C$:
  \[
    \pi^\dagger_!\ObjTerm{}
    \cong
    \pi^\dagger_!\prn{\Colim{\Modify[\OM]{\TotalType{X}}} \Yo}
    \cong
    \Colim{\Modify[\OM]{\TotalType{X}}} \pi^\dagger_! \circ \Yo
    \cong
    \Colim{\Modify[\OM]{\TotalType{X}}} \Yo \circ \pi^\dagger
  \]
  Ми використали той факт, що лівий спряжений відображення $\pi^\dagger_!$ комутує з кограницями~\parencite{bardomiano:2021}. Щоб показати, що $\pi^\dagger_!\ObjTerm{} \cong X$, ми зауважимо, що для всіх $Z : \PSH{C}$:
  \begin{align*}
    \Hom{\pi^\dagger_!\ObjTerm{}}{Z}
    &\Equiv \Hom[{\TotalType{X}} \to \Space]{\ObjTerm{}}{Z \circ \pi}
    \\
    &\Equiv \Prod{\prn{c, x} : \Sum{c : \Modify[\OM]{C}} X\prn{c}} Z\prn{c}
    \\
    &\Equiv \Prod{c : \Modify[\OM]{C}} X\prn{c} \to Z\prn{c}
    \\
    &\Equiv \Hom{X}{Z}
  \end{align*}
  Тепер висновок випливає з леми Йонеди.
\end{proof}

\begin{lem}
  $\Nv_f = \Phi\prn{f^\dagger\prn{-},-} : E \to \PSH{C}$ є правим спряженим.
\end{lem}
\begin{proof}
  Ми доведемо, що відображення $\Nv_f$ є поточково правим спряженим. Відповідно, фіксуючи $\DeclVar{X}{\GM}{\PSH{C}}$, ми маємо побудувати $\DeclVar{e}{}{\Modify[\OM]{E}}$, таке що $\Phi\prn{e,-} \cong \Phi\prn{\MkMod[\OM]{X},\Nv_f\prn{-}}$. Оскільки $X \cong \Colim{\Modify[\OM]{\TotalType{X}}} \Yo \circ \pi^\dagger$ і категорія $E$ є коповною, за принципом дуальності до Наслідку~\ref{cor:adj:yo-preserves-limits} достатньо припустити, що $\MkMod[\OM]{X} = \Yo^\dagger\prn{c}$, де $\DeclVar{c}{}{\Modify[\OM]{C}}$.\footnote{Зауважте відсутність $\GM$-анотації тут: ми повинні гарантувати, що ми є функторіальними за $c$, щоб отримати \emph{діаграму} в $E$.} Зрештою, покладемо $e = f^\dagger\prn{c}$, тоді $\Phi\prn{f^\dagger\prn{c},-} \cong \Phi\prn{\Yo^\dagger\prn{c},\Nv_f\prn{-}}$ за Теоремою~\ref{thm:tw:strong-yoneda}.
\end{proof}

Тепер ми можемо довести, як і обіцяли, універсальну властивість категорії $\PSH{C}$. Якщо ми запишемо $\CoCont{\PSH{C}}{E}$ для повної підкатегорії функторів $\PSH{C} \to E$, які зберігають усі кограниці, то відображення $\Yo^* : \CoCont{\PSH{C}}{E} \to \prn{C \to E}$ є еквівалентністю. Щоб довести це, ми, по суті, доводимо, що існує відображення, яке переводить $f$ у лівий спряжений до $\Nv_f$, і що воно є оберненим до $\Yo^*$.

\begin{restatable}{thm}{universalpropofpsh}
  \label{thm:adj:cc}
  $\Yo^* : \CoCont{\PSH{C}}{E} \to \prn{C \to E}$ є еквівалентністю.
\end{restatable}
\begin{proof}
  Ми використовуємо Теорему~\ref{thm:prelims:ff-ess-surj-to-equiv}. Якщо $\DeclVar{f}{\GM}{C \to E}$, то відображення $f_! : \PSH{C} \to E$ задовольняє рівність $f_! \circ \Yo = f$, а отже $\Yo^*$ є істотно сюр'єктивним:
  \[
    \Phi\prn{\prn{f_!\circ \Yo}^\dagger\prn{-},-} \cong \Phi\prn{\Yo^\dagger\prn{-},\Nv_f\prn{-}}
    \cong \Nv_f = \Phi\prn{f^\dagger\prn{-},-}
  \]

  Більше того, якщо $\DeclVar{F}{\GM}{\CoCont{\PSH{C}}{E}}$, то $\prn{F\circ\Yo}_! \cong F$, так що $\Yo^*$ є бієкцією на $\GM$-елементах. Запишемо $f = F \circ \Yo$. Ми спочатку побудуємо порівняльне відображення (comparison map) $\Hom{f_!}{F}$, побудувавши природне перетворення $\Hom{\ArrId{}}{\Nv_{f}\prn{F\prn{-}}}$. За карруванням (currying), це еквівалентно побудові природного перетворення між відображеннями $\Modify[\OM]{C} \times \PSH{C} \to \Space$, і в такій формі тотожне відображення $\ArrId{}$ задається оцінкою $\epsilon$, а $\Nv_F\prn{F\prn{-}}$ дорівнює $\Phi\prn{f\prn{-},F\prn{-}}$. Ми можемо замінити $\epsilon$ на $\Phi\prn{\Yo\prn{-},-}$ за Теоремою~\ref{thm:tw:strong-yoneda}, а $\Phi\prn{f\prn{-},F\prn{-}} = \Phi\prn{F\prn{\Yo\prn{-}}, F\prn{-}}$ за визначенням. Відповідно, потрібне відображення забезпечується природністю $\Phi_F$. Досить легко перевірити, що воно є поточково еквівалентністю за Лемою~\ref{lem:adj:density}.

  Зрештою, тепер ми покажемо, що відображення $\Yo^*$ є строго повним. Для цього ми маємо показати, що якщо $\DeclVar{f,g}{\GM}{C \to E}$, то $\Hom{f_!}{g_!} \cong \Hom{f}{g}$. Обидві сторони є групоїдами, тому достатньо розглянути $\GM$-анотовані елементи. Якщо $\DeclVar{\alpha}{\GM}{\Hom{f}{g}}$, то за допомогою транспонування ми можемо розглядати $\alpha$ як елемент типу $\Modify[\GM]{C \to E^\Int}$, і попереднє спостереження гарантує, що цей тип є еквівалентним $\Modify[\GM]{\CoCont{\PSH{C}}{E^\Int}}$, що й дає бажаний висновок після транспонування.
\end{proof}
\section{Теорія розширень Кана}
\label{sec:kan}
Об'єднуючим поняттям у теорії категорій є \emph{розширення Кана} (Kan extensions), які є універсальними розширеннями функторів уздовж функторів на тій самій області визначення. Маклейн, один із засновників теорії категорій, відомо зазначив: «Поняття розширення Кана охоплює всі інші фундаментальні поняття теорії категорій», такі як (ко)границі та спряження~\parencite{maclane:1978,riehl:2014}.

\begin{defi}[Розширення Кана]
  Для заданого відображення $f : C \to D$ та категорії $E$, ліве (праве) розширення Кана $\Lan{f}$ ($\Ran{f}$) є лівим (правим) спряженим до відображення передкомпозиції $f^* : E^D \to E^C$.
\end{defi}

Хоча це визначення має сенс у загальному випадку, для використання результатів попередніх розділів ми будемо припускати, що $\DeclVar{f}{\GM}{C \to D}$ та $\DeclVar{E}{\GM}{\Uni}$. У Підрозділі~\ref{sec:kan:existence} ми покажемо, що розширення Кана існують завжди, коли $E$ є (ко)повною, а у Підрозділах~\ref{sec:kan:cofinal} та~\ref{sec:kan:quillen-a} ми застосуємо це на практиці, вивівши два важливі результати: теорему А Квіллена та правильність (properness) кодекартових розшарувань. Наші аргументи щодо існування розширень Кана та теореми А Квіллена адаптують (модельно-незалежні) $\infty$-категорні міркування з \textcite{ramzi:bousfield-kan}.

\subsection{Існування та описання розширень Кана}
\label{sec:kan:existence}

Ми можемо довести, що розширення Кана можуть бути обчислені очікуваним способом. Для $\DeclVar{d}{}{D}$ ми пишемо $\SLICE{C}{d} \defeq C \times_D \SLICE{D}{d}$ та $\COSLICE{C}{d} \defeq C \times_D \COSLICE{D}{d}$.
Ми припускаємо, що $C$ та $D$ обидві є малими категоріями, тому кожна $\SLICE Cd$ також є малою.
За Теоремою~\ref{thm:adj:precomp-is-right-adj} та Лемою~\ref{lem:adj:comp}:
\begin{lem}
  \label{lem:kan:lan-exists-for-psh}
  Якщо $E = \PSH{A}$ для деякої категорії $\DeclVar A\GM\Uni$, тоді $\Lan{f}$ існує.
  Більше того, якщо $\DeclVar{X}{\GM}{C \to E}$ та $\DeclVar{d}{\GM}{D}$, то \\
  $\Lan{f}\,X\,d = \Colim{} \prn{\SLICE{C}{d} \to C \to E} = \Grpdify\prn{\Sum{\prn{c,\_} : \SLICE{C}{d}} X\prn{c}}$.
\end{lem}
Це дає більш загально:
\begin{thm}
  \label{thm:kan:left-kan-exists}
  Якщо $E$ є коповною категорією, то $\Lan{f}$ існує, і якщо $\DeclVar{X}{\GM}{C \to E}$, $\DeclVar{d}{\GM}{D}$, то $\Lan{f}\,X\,d = \Colim{} \prn{\SLICE{C}{d} \to C \to E}$.
\end{thm}
\begin{proof}
  Достатньо довести, що передкомпозиція є поточково правим спряженим, і тому ми фіксуємо $\DeclVar{X}{\GM}{C \to E}$. За Теоремою~\ref{thm:adj:cc} ми можемо розглядати $X$ як композицію $\bar X \circ \Yo$, де $\bar X : \PSH{C} \to E$ є лівим спряженим до $\Nv_X$.
  Далі, за Лемою~\ref{lem:kan:lan-exists-for-psh} ми зауважуємо, що $\Yo: C \to \PSH C$ допускає розширення на $D$ уздовж $f$, а саме $\Lan{f} \Yo : D \to \PSH C$, і ми стверджуємо, що $\bar{X} \circ \Lan{f} \Yo$ є нашим шуканим розширенням для $f$.
  Фіксуючи $Z : D \to E$, обчислимо:
  \begin{align*}
    \Hom[D \to E]{\bar{X} \circ \Lan{f} \Yo}{Z}
    &\Equiv \Hom[D \to \PSH{C}]{\Lan{f} \Yo}{\Nv_X \circ Z}
    \\
    &\Equiv \Hom[C \to \PSH{C}]{\Yo}{\Nv_X \circ Z \circ f}
    \\
    &\Equiv \Hom[C \to E]{\bar X\circ\Yo}{Z \circ f}
    \\
    &= \Hom[C \to E]{X}{Z \circ f}
  \end{align*}
  Очікувана формула кограниці продовжує виконуватися як наслідок Леми~\ref{lem:kan:lan-exists-for-psh} та конеперервності (cocontinuity) $\bar{X}$.
\end{proof}

Завдяки дуальності ми отримуємо наступний варіант:
\begin{thm}
  \label{thm:kan:right-kan-exists}
  Якщо $E$ є повною категорією, то $\Ran{f}$ існує і задається дуальною формулою границі:
  $\Ran{f}\,X\,d = \Lim{} \prn{\COSLICE{C}{d} \to C \to E}$.
\end{thm}

\subsection{Фінальні та ініціальні функтори}
\label{sec:kan:cofinal}

Часто корисно показати, що границю складної діаграми $D$ можна обчислити, спочатку звузивши її до простішої діаграми за допомогою відображення $f : C \to D$ та обчисливши границю там, наприклад, звужуючи з $\mathbb{Z}$ до $\mathbb{Z}_{\le 0}$. Коли такий підхід є правильним, відображення $f$ називають ініціальним (initial):
\begin{defi}
  Функтор $\DeclVar{f}{\GM}{C \to D}$ є \emph{ініціальним}, якщо для кожного $\DeclVar{X}{\GM}{D \to \Space}$ відображення $\Lim{D} X \to \Lim{C} X \circ f$ є еквівалентністю. Відображення є \emph{фінальним} (final), якщо двоїсте до нього відображення є ініціальним.
\end{defi}
Хоча це визначення є асиметричним у своєму трактуванні ініціальності та фінальності, ми відновимо симетрію як наслідок теореми А Квіллена у наступному підрозділі, див.~Наслідок~\ref{cor:kan:right-cofinal-char}.

Нагадаємо, що $\Lim{D} X = \Prod{d : D} X\prn{d}$, тому визначення ініціальності еквівалентно стверджує, що відображення звуження $\prn{\Prod{d : D} X\prn{d}} \to \prn{\Prod{c : C} X\prn{f\prn{c}}}$ є еквівалентністю.

\begin{exa}
  Вкладення $\brc{0}$/$\brc{1}$ виду ${\Unit \to \Int}$ є ініціальним/фінальним.
\end{exa}

\begin{lem}
  \label{lem:kan:left-cofinal-preserves-limit}
  Якщо відображення $\DeclVar{f}{\GM}{C \to D}$ є ініціальним і $\DeclVar{X}{\GM}{D \to E}$, тоді $\Lim{C} \prn{X \circ f}$ та $\Lim{D} X$ обидва існують, якщо існує хоча б одне з них, і є канонічно ізоморфними.
\end{lem}
\begin{proof}
  За Наслідком~\ref{cor:adj:yo-preserves-limits} ми замінюємо $E$ на $\PSH{E}$, а за Наслідком~\ref{cor:adj:limits-are-pointwise} зводимо задачу до $\Space$, де результат є безпосереднім.
\end{proof}

\begin{restatable}{lem}{locop}
  \label{lem:app:localization-op}
  Якщо $\DeclVar{C}{\GM}{\Uni}$, тоді $\Grpdify{C} \Equiv \Grpdify{\Modify[\OM]{C}}$.
\end{restatable}
\begin{proof}
  Ми зауважуємо, що $\Grpdify{C} \Equiv \Modify[\GM]{\Grpdify{C}}$ і аналогічно для $\Modify[\OM]{C}$. Відповідно, ми маємо:
  \begin{gather*}
    \Modify[\GM]{
      \Modify[\OM]{C} \to \Modify[\GM]{X}
    }
    \Equiv
    \Modify[\GM]{
      \Grpdify{\Modify[\OM]{C}} \to \Modify[\GM]{X}
    }
    \\
    \Modify[\GM]{
      C \to \Modify[\GM]{X}
    }
    \Equiv
    \Modify[\GM]{
      \Grpdify{C} \to \Modify[\GM]{X}
    }
    \\
    \Modify[\GM]{
      C \to \Modify[\GM]{X}
    }
    \Equiv
    \Modify[\GM]{
      \Modify[\OM]{C} \to \Modify[\GM]{X}
    }
  \end{gather*}
  Зрештою, результат випливає з простого міркування Йонеди.
\end{proof}

\begin{lem}
  \label{lem:kan:localization-is-cofinal}
  Для кожної категорії $C$ канонічне відображення $C \to \Grpdify{C}$ є як ініціальним, так і фінальним.
\end{lem}
\begin{proof}
  За Лемою~\ref{lem:app:localization-op} достатньо довести, що це відображення є ініціальним. Для цього ми повинні показати, що наступне відображення є еквівалентністю для кожного $X : \Grpdify{C} \to \Space$:
  \[
    \prn{\Prod{d : \Grpdify{C}} X\prn{d}} \to \prn{\Prod{c : C} X\prn{\eta\prn{c}}}
  \]
  Проте $X\prn{d}$ є дискретним для кожного $d : \Grpdify{C}$, тому це просто універсальна властивість $\Grpdify$.
\end{proof}
\begin{cor}
  \label{lem:kan:limit-over-contractible}
  Якщо $\Grpdify{C} = \ObjTerm{}$, тоді $\Lim{C} A = A$ для $A : \Space$.
\end{cor}

\subsection{Теорема А Квіллена}
\label{sec:kan:quillen-a}

Нашою наступною метою є доведення $\infty$-категорної версії теореми А Квіллена. На відміну від традиційних доведень, ми слідуємо \textcite{ramzi:bousfield-kan} і спираємося на вже розроблений базовий апарат розширень Кана, щоб спростити наші міркування.

\begin{defi}
  Функтор $\DeclVar{f}{\GM}{C \to D}$ є фінальним за Квілленом (Quillen final), якщо $\Grpdify(\COSLICE{C}{d}) \Equiv \Unit$ для всіх $\DeclVar{d}{\GM}{D}$.
\end{defi}
\begin{thm}
  \label{thm:kan:theorem-a}
  Функтор $\DeclVar{f}{\GM}{C \to D}$ є фінальним тоді й лише тоді, коли він є фінальним за Квілленом.
\end{thm}

\begin{rem}
  Цей результат показує, зокрема, що фінальність не залежить від обраного конкретного універсуму $\Space$.
\end{rem}

\begin{lem}
  \label{lem:kan:theorem-a-for-psh}
  Якщо відображення $\DeclVar{f}{\GM}{C \to D}$ є фінальним за Квілленом і $\DeclVar{X}{\GM}{D \to \PSH{A}}$, то ${\Colim D} X \Equiv {\Colim C} X \circ f$.
\end{lem}
\begin{proof}
  Це твердження є поточковим, тому ми швидко зводимо його до випадку $\Space$ замість $\PSH{A}$. У цій ситуації ми хочемо показати, що наступна діаграма комутує:
  \[
    \begin{tikzcd}
      \Space^D \ar[rr,"f^*"]\ar[dr,"{\mathrm{colim}_D}"'] && \Space^C\ar[dl,"{\mathrm{colim}_C}"] \\
      & \Space &
    \end{tikzcd}
  \]
  Зауважимо, що всі три морфізми є лівими спряженими, а тому достатньо порівняти їхні праві спряжені: постійні функтори $\Delta_C$ та $\Delta_D$, разом із правим розширенням Кана $\Ran f$.
  Далі ми помічаємо, що існує щонайменше порівняльне відображення $\Delta_D \to \Ran{f} \circ \Delta_C$, задане транспонуванням тотожного відображення $f^* \circ \Delta_D \to \Delta_C$.
  Ми маємо довести, що це відображення є поточково оборотним, тому зводимо задачу до розгляду $\DeclVar{X}{\GM}{\Space}$ та $\DeclVar{d}{\GM}{D}$, і повинні показати наступне, використовуючи~Теорему~\ref{thm:kan:right-kan-exists}:
  $X \Equiv \Lim{\COSLICE{C}{d}} X$.
  Це випливає з нашого припущення та~Леми~\ref{lem:kan:limit-over-contractible}.
\end{proof}

\begin{lem}
  \label{lem:kan:theorem-a}
  Якщо відображення $\DeclVar{f}{\GM}{C \to D}$ є фінальним за Квілленом, $E$ --- коповна категорія, і $\DeclVar{X}{\GM}{D \to E}$, тоді ${\Colim D} X \Equiv {\Colim C} X \circ f$.
\end{lem}
\begin{proof}
  Ми зводимо задачу до випадку $E = \PSH{D}$ (і, отже, до Леми~\ref{lem:kan:theorem-a-for-psh}), розкладаючи $X$ як $\bar{X} \circ \Yo$ і зауважуючи, що відображення $\bar{X}$ зберігає кограниці за побудовою.
\end{proof}

\begin{proof}[Доведення Теореми~\ref{thm:kan:theorem-a}]
  Щоб побачити, що фінальність за Квілленом тягне за собою фінальність, застосуємо Лему~\ref{lem:kan:theorem-a} до $\Modify[\OM]{\Space}$ і обчислимо:
  \[
    \Lim{\Modify[\OM]D} X
    \Equiv {\Colim D} X^\dagger
    \Equiv {\Colim C} X^\dagger \circ f
    \Equiv \Lim{\Modify[\OM]C} X \circ f^\dagger
  \]
  Для зворотного твердження припустимо, що відображення $f$ є фінальним. Ми помічаємо, що за принципом дуальності до Леми~\ref{lem:kan:left-cofinal-preserves-limit} (знову застосованої до $\Modify[\OM]{\Space}$) канонічне відображення $\Colim{C} X \circ f \to \Colim{D} X$ є еквівалентністю для будь-кого $\DeclVar{X}{\GM}{D \to \Space}$. Фіксуємо $\DeclVar d\GM D$ і виберемо $X = \hom(d,-) = \Phi\prn{\MkMod[\OM]d,-}$, таке що розглядувані кограниці є саме $\Grpdify{\COSLICE{D}{d}}$ та $\Grpdify{\COSLICE{C}{d}}$, використовуючи~Теорему~\ref{thm:adj:precomp-is-right-adj}. Це завершує доведення, оскільки $\Grpdify{\COSLICE{D}{d}} = \ObjTerm{}$.
\end{proof}

\begin{cor}
  \label{cor:kan:right-cofinal-char}
  Функтор $\DeclVar{f}{\GM}{C \to D}$ є фінальним тоді й лише тоді, коли для кожного $\DeclVar{X}{\GM}{D \to \Space}$ відображення $\Colim{D} X \to \Colim{C} X \circ f$ є еквівалентністю.
\end{cor}
Це відновлює симетрію між ініціальними та фінальними функторами, як і було обіцяно. Ми пропонуємо ще одне симетричне визначення ініціальності та фінальності, спираючись на \textcite[Розд.~8]{cisinski:2024}.
\begin{defi}[Коваріантні еквівалентності]
  Фіксуємо $\DeclVar{p}{\GM}{C \to A}$ та $\DeclVar{q}{\GM}{D \to A}$ між категоріями $A,C,D$. Нехай $\DeclVar{f}{\GM}{C \to D}$ є шаруватим відображенням (fibered map) наступного вигляду:
  \[
    \begin{tikzcd}
     C && D \\
     & A
     \arrow["f", from=1-1, to=1-3]
     \arrow["p"', from=1-1, to=2-2]
     \arrow["q", from=1-3, to=2-2]
    \end{tikzcd}
  \]
  Ми називаємо $f$ \emph{коваріантною еквівалентністю}, якщо для всіх сімейств $\DeclVar{X}{\GM}{A \to \Space}$ реіндексування (reindexing) породжує еквівалентність, тобто:
  \[
    f^* : \prn{\DelimMin{1}\Prod{a:A} D_a \to X_a} \to \prn{\DelimMin{1}\Prod{a:A} C_a \to X_a}
  \]
  Дуально, відображення $f$ називається \emph{контраваріантною еквівалентністю}, якщо передкомпозиція відносно всіх контраваріантних сімейств є еквівалентністю.
\end{defi}

\begin{lem}\label{lem:cov-eq-comp}
  Нехай відображення $f$, як показано нижче, є коваріантною еквівалентністю відносно $p$ та $q$. Тоді для будь-якого функтора $\DeclVar{r}{\GM}{B \to A}$ воно також є коваріантною еквівалентністю відносно $rp$ та $rqbw$:
  \[
    \begin{tikzcd}
      C && D \\
      & A \\
      & B
      \arrow["f", from=1-1, to=1-3]
      \arrow["p"', from=1-1, to=2-2]
      \arrow["q", from=1-3, to=2-2]
      \arrow["r", from=2-2, to=3-2]
    \end{tikzcd}
  \]
\end{lem}

\begin{proof}
  Ми отримуємо наступний індукований квадрат:
  \[
    \begin{tikzcd}
      {\big( \prod_{b:B} D_b \to X_b \big)} && {\big( \prod_{b:B} C_b \to X_b \big)} \\
      {\big( \prod_{a:A} D_a \to X_{r(a)} \big)} && {\big( \prod_{a:A} C_a \to X_{r(a)} \big)}
      \arrow["{f^*}", "\simeq"', from=1-1, to=1-3]
      \arrow["\simeq"', from=1-1, to=2-1]
      \arrow["\simeq", from=1-3, to=2-3]
      \arrow["{f^*}"', from=2-1, to=2-3]
    \end{tikzcd}
  \]
  Верхнє горизонтальне відображення є еквівалентністю за передумовами. Мета полягає в тому, щоб показати, що нижнє горизонтальне відображення також є еквівалентністю. Але це випливає з властивості «3-з-2» для еквівалентностей.
\end{proof}

\begin{lem}[Характеризації ініціальності]
  \label{lem:kan:characterization-initiality}
  Нехай $\DeclVar{f}{\GM}{C \to D}$ --- функтор. Тоді наступні умови є еквівалентними:
  \begin{enumerate}[label=\textup{(\arabic*)}]
  \item\label{it:prop:char-left-cof-i} відображення $f$ є ініціальним.
  \item\label{it:prop:char-left-cof-ii} Нехай $\DeclVar{X}{\GM}{A \to \Space}$ --- сімейство з асоційованим лівим розшаруванням $\DeclVar{\pi}{\GM}{\TotalType{X} \to A}$. Тоді будь-який квадрат наступного вигляду має наповнення (filler) $\overline{\varphi}$, єдине з точністю до гомотопії:
    \[
      \begin{tikzpicture}
        \node (nw) {$C$};
        \node[below = 1.5cm of nw] (sw) {$D$};
        \node[right = 4cm of nw] (ne) {$\TotalType{X}$};
        \node[right = 4cm of sw] (se) {$A$};
        \path[->] (nw) edge node[above] {$\varphi$} (ne);
        \path[->] (nw) edge node[left] {$f$} (sw);
        \path[->] (ne) edge node[right] {$\pi$} (se);
        \path[->] (sw) edge node[below] {$\alpha$} (se);
        \path[->, unique exists] (sw) edge node[upright desc] {$\overline{\varphi}$} (ne);
      \end{tikzpicture}
    \]
  \item\label{it:prop:char-left-cof-iii} Для будь-кого сімейства $\DeclVar{X}{\GM}{A \to \Space}$ наступний квадрат є декартовим (pullback):
    \[
      \DiagramSquare{
        nw = \TotalType{X}^D,
        ne = \TotalType{X}^C,
        sw = A^D,
        se = A^C,
        nw/style = pullback,
        height = 1.5cm,
      }
    \]
  \item\label{it:prop:char-left-cof-iv} відображення $f$ є коваріантною еквівалентністю відносно будь-якого $\DeclVar{\alpha}{\GM}{D \to A}$.
  \end{enumerate}
  Аналогічна характеристика виконується для контраваріантних еквівалентностей та фінальних функторів.
\end{lem}
\begin{proof}
  Легко бачити, що умови~\ref{it:prop:char-left-cof-ii} та~\ref{it:prop:char-left-cof-iii} є еквівалентними завдяки комутуванню $\Prod{}$ та $\Sum{}$. Умова~\ref{it:prop:char-left-cof-iv} розгортається у наступне твердження: для будь-якого $\DeclVar{X}{\GM}{A \to \Space}$ реіндексування вздовж $f$ є еквівалентністю, а саме:
  \[
    f^* :
    \prn{\Prod{a:A} D_a \to X_a}
    \xrightarrow{\sim}
    \prn{\Prod{a:A} (\Sum{d:D_a} C_{a,d}) \to X_a}
  \]
  Це, знову ж таки, легко бачити еквівалентним до~\ref{it:prop:char-left-cof-ii}.

  Перейдемо до імплікації $\text{\ref{it:prop:char-left-cof-iv}} \implies \text{\ref{it:prop:char-left-cof-i}}$. Але це очевидно, оскільки~\ref{it:prop:char-left-cof-i} стверджує, що $f$ є коваріантною еквівалентністю відносно самого себе та $\ArrId{D}$.

  Для зворотного напрямку $\text{\ref{it:prop:char-left-cof-i}} \implies \text{\ref{it:prop:char-left-cof-iv}}$ ми використовуємо щойно отримане розуміння разом із Лемою~\ref{lem:cov-eq-comp}.
\end{proof}

Наступна альтернативна характеристика також часто є корисною:

\begin{restatable}{lem}{charcofinal}
  \label{prop:kan:charcofinal}
  $\DeclVar{f}{\GM}{C \to D}$ є ініціальним \textup(відп. фінальним\textup) тоді й лише тоді, коли для кожного коваріантного \textup(відп. контраваріантного\textup) сімейства $\DeclVar{\pi}{\GM}{X \to Y}$ відображення $f$ є ліво-ортогональним до $\pi$, тобто $\IsEquiv\prn{X^D \to X^C \times_{Y^C} Y^D}$.
\end{restatable}
\begin{proof}
  Безпосередньо випливає з Твердження~\ref{it:prop:char-left-cof-iv} для ініціального випадку, та з дуальності й~Наслідку~\ref{cor:kan:right-cofinal-char} для фінального випадку.
\end{proof}

\begin{cor}
  \label{cor:initial-covariant-ofs}
  Існує ортогональна система факторизації (orthogonal factorization system) у сенсі \cite{rijke:2020}, лівий клас якої задається ініціальними функторами, а правий --- коваріантними розшаруваннями.
\end{cor}

Як ще один наслідок ми отримуємо дуальне твердження до~Теореми~\ref{thm:kan:theorem-a}:
\begin{cor}
  \label{thm:kan:theorem-a-dual}
  Функтор $\DeclVar{f}{\GM}{C \to D}$ є ініціальним тоді й лише тоді, коли $\Grpdify(\SLICE{C}{d}) \Equiv \Unit$ для всіх $\DeclVar{d}{\GM}{D}$ \textup(\emph{ініціальний за Квілленом}\textup).
\end{cor}

Ми демонструємо корисність Теореми~\ref{thm:kan:theorem-a}, наводячи нове і набагато простіше доведення того, що кодекартові розшарування є \emph{правильними} (proper).
\begin{defi}
  \label{def:proper-maps}
  Функтор $\DeclVar{\pi}{\GM}{E \to B}$ між категоріями є \emph{правильним}, якщо для всіх декартових квадратів (pullbacks) ($\GM$-функторів) наступного вигляду відображення $v$ є фінальним, якщо $u$ є фінальним:
  \[
    \begin{tikzpicture}[diagram]
      \SpliceDiagramSquare{
        height = 1.35cm,
        ne = E,
        se = B,
        nw = E',
        sw = B',
        east = \pi,
        west = \pi',
        nw/style = pullback,
      }
      \node [pullback, left = of nw] (A) {$E''$};
      \node [left = of sw] (B) {$B''$};
      \path[->] (A) edge node[above] {$v$} (nw);
      \path[->] (B) edge node[below] {$u$} (sw);
      \path[->] (A) edge (B);
    \end{tikzpicture}
  \]
  Ми називаємо відображення $\pi$ \emph{гладким} (smooth), якщо $\pi^\dagger : \Modify[\OM]{E} \to \Modify[\OM]{B}$ є правильним.
\end{defi}
\begin{lem}
  Гладкі та правильні функтори є замкненими відносно композиції та взяття декартового квадрата (pullback).\footnote{Визначення правильності сформульовано спеціально так, щоб вбудувати останню властивість.}
\end{lem}
\begin{thm}
  \label{thm:kan:cocartesian-is-proper}
  Декартові розшарування є гладкими, а кодекартові розшарування є правильними.
\end{thm}
\begin{proof}
  Достатньо розглянути правильний випадок. Фіксуємо кодекартове розшарування $\DeclVar{\pi}{\GM}{E \to B}$ і зауважимо, що оскільки кодекартові розшарування є стійкими відносно декартових квадратів, достатньо показати, що $v$ є фінальним у наступній діаграмі декартового квадрата, якщо $u$ є фінальним:
  \[
    \DiagramSquare{
      height = 1.5cm,
      width = 3cm,
      nw = { A \times_B E},
      sw = A,
      ne = E,
      se = B,
      nw/style = pullback,
      north = v,
      south = u,
      east = \pi,
    }
  \]
  Тепер ми використовуємо Теорему~\ref{thm:kan:theorem-a}. Для $\DeclVar{e}{\GM}{E}$ ми обчислюємо шар:
  \begin{align*}
    & (A \times_B E) \times_E \COSLICE{E}{e}
    \\
    &\quad
    \Equiv A \times_B \COSLICE{E}{e}
    \\
    &\quad
    \Equiv A \times_B \prn{\Sum{\DeclVar{b'}{}{B},\DeclVar{f}{}{\Hom{\pi\prn{e}}{b'}}}{\prn{E_{b'}}^{\Int}}}
    && \text{$\pi$ є кодекартовим}
    \\
    &\quad
    \Equiv \Sum{\DeclVar{(a,f)}{}{A \times_B \COSLICE{B}{\pi\prn{e}}}}{\COSLICE{\prn{E_{u\prn{a}}}}{f_!e}}
  \end{align*}
  Застосування $\Grpdify$ до кожного шару дає $\Grpdify{\COSLICE{\prn{E_{u\prn{a}}}}{f_!e}} \Equiv \Unit$ (оскільки козрізи (coslices) мають початкові елементи) та $\Grpdify(A \times_B \COSLICE{B}{\pi\prn{e}}) \Equiv \Unit$, оскільки відображення $u$ є фінальним за припущенням. Це означає, що застосування $\Grpdify$ до всього $\Sum{}$-типу дає $\Unit$~\parencite{rijke:2020}.
\end{proof}

\begin{cor}
  Якщо відображення $\DeclVar{\pi}{\GM}{E \to B}$ є кодекартовим і $\DeclVar{X}{\GM}{E \to D}$, то ліве розширення Кана $\Lan{\pi} X$ переводить $\DeclVar{b}{\GM}{B}$ у $\Colim{}\prn{E_b \to E \to D}$.
\end{cor}

\subsection{Гладка та правильна зміна бази}

Ми хочемо показати, що гладкі та правильні функтори задовольняють \emph{умову Бека--Шевальє} (Beck--Chevalley condition). Ми слідуємо \cite[Розділ 8.4]{cisinski:2024}; див. також \cite{anel:2024} для загального обговорення.

По-перше, використовуючи ініціально-коваріантну факторизацію (див. \ref{cor:initial-covariant-ofs}) функтора з малими шарами, ми можемо обчислити дію передкомпозиції для (ко)передпучків:
\begin{prop}[\protect{\cite[Теорема~8.1.18]{cisinski:2024}}]
  \label{prop:action-postcomp}
  Нехай $\DeclVar{u}{\GM}{A \to B}$ --- функтор із малими шарами. Тоді лівий спряжений $\DeclVar{u_! \dashv u^*}{\GM}{\Space^B \to \Space^A}$ діє наступним чином: для $\DeclVar{F}{\GM}{A \to \Space}$, якщо
  \[\begin{tikzcd}
	E & {\Space_*} \\
	A & \Space
	\arrow[from=1-1, to=1-2]
	\arrow["\phi"', from=1-1, to=2-1]
	\arrow["\lrcorner"{anchor=center, pos=0.125}, draw=none, from=1-1, to=2-2]
	\arrow[from=1-2, to=2-2]
	\arrow["F", from=2-1, to=2-2]
  \end{tikzcd}\]
  позначимо через $E \stackrel{j}{\to} X \stackrel{\psi}{\to} B$ ініціально-коваріантну факторизацію відображення $u \circ \phi$. Тоді $u_!(F)$ є розпрямленням (straightening) для $\psi$:
  \[\begin{tikzcd}
	E & X & {\Space_*} \\
	A & B & \Space
	\arrow["j", from=1-1, to=1-2]
	\arrow["\phi"', from=1-1, to=2-1]
	\arrow[from=1-2, to=1-3]
	\arrow["\psi", from=1-2, to=2-2]
	\arrow["\lrcorner"{anchor=center, pos=0.125}, draw=none, from=1-2, to=2-3]
	\arrow[from=1-3, to=2-3]
	\arrow["u"', from=2-1, to=2-2]
	\arrow["{u_!(F)}"', from=2-2, to=2-3]
\end{tikzcd}\]
Зокрема, одиниця $\DeclVar{\eta}{\GM}{F \to u^*u_!F}$ відповідає за спрямованою одновалентністю (directed univalence) відображенню $\widetilde{\eta}$:
\[\begin{tikzcd}
	E & Y & X \\
	& A & B
	\arrow["{\widetilde{\eta}}", dashed, from=1-1, to=1-2]
	\arrow["j"{description}, curve={height=-20pt}, from=1-1, to=1-3]
	\arrow["\phi"{description}, from=1-1, to=2-2]
	\arrow[from=1-2, to=1-3]
	\arrow["{u^*\psi}", from=1-2, to=2-2]
	\arrow["\lrcorner"{anchor=center, pos=0.125}, draw=none, from=1-2, to=2-3]
	\arrow["\psi", from=1-3, to=2-3]
	\arrow["u", from=2-2, to=2-3]
\end{tikzcd}\]
\end{prop}

\begin{proof}
  Ми хочемо показати, що відображення
  \[ \DeclVar{\lambda \alpha.u^*\alpha \circ \eta}{\GM}{\hom_{B \to \Space}(u_!F, G) \to \hom_{A \to \Space}(F, u^*G)}\]
  задає еквівалентність. Ми запишемо корозпрямлення (unstraightening) для $G$ як:
  \[\begin{tikzcd}
	Z & {\Space_*} \\
	B & \Space
	\arrow[from=1-1, to=1-2]
	\arrow["\gamma"', from=1-1, to=2-1]
	\arrow["\lrcorner"{anchor=center, pos=0.125}, draw=none, from=1-1, to=2-2]
	\arrow[from=1-2, to=2-2]
	\arrow["G", from=2-1, to=2-2]
  \end{tikzcd}\]
  Наведене вище відображення транспонування є еквівалентним відображенню
  \[ \prn{\prod_{b:B} \hom_\Space(X_b, Z_b)} \to \prn{\prod_{a:A} \hom_\Space(E_a, (A \times_B Z)_a)} \simeq \prn{\prod_{b:B} \hom_\Space(E_b,Z_b)}.\]
  За спрямованою одновалентністю, це відображення відповідає відображенню
  \[ \prn{\prod_{b:B} (X_b \to Z_b)} \to\prn{\prod_{b:B} (E_b \to Z_b)} \]
  яке, у свою чергу, діє як передкомпозиція шаруватих функторів відображенням $j$:
  \[\begin{tikzcd}
	E & X & Z \\
	& B
	\arrow["j", from=1-1, to=1-2]
	\arrow["{u \circ \varphi}"', from=1-1, to=2-2]
	\arrow[from=1-2, to=1-3]
	\arrow["\psi"', from=1-2, to=2-2]
	\arrow["\gamma", from=1-3, to=2-2]
\end{tikzcd}\]
Зрештою, в силу Леми~\ref{prop:kan:charcofinal}\ref{it:prop:char-left-cof-iv} це відображення є еквівалентністю.
\end{proof}

Нехай $\DeclVar{f}{\GM}{A \to B}$ --- функтор із малими шарами. Розглянемо індуковану пару спряжених відображень:
\[\begin{tikzcd}
	{\Space^B} & {\Space^A}
	\arrow[""{name=0, anchor=center, inner sep=0}, "{f^*}"', curve={height=12pt}, from=1-1, to=1-2]
	\arrow[""{name=1, anchor=center, inner sep=0}, "{f_!}"', curve={height=12pt}, from=1-2, to=1-1]
	\arrow["\dashv"{anchor=center, rotate=-90}, draw=none, from=1, to=0]
\end{tikzcd}\]
Таким чином, квадрат
\[\begin{tikzcd}
	{A'} & A \\
	{B'} & B
	\arrow["u", from=1-1, to=1-2]
	\arrow["{f'}"', from=1-1, to=2-1]
	\arrow["f", from=1-2, to=2-2]
	\arrow["v"', from=2-1, to=2-2]
\end{tikzcd}\]
індукує \emph{квадрат Бека--Шевальє} (Beck--Chevalley square)
\[\begin{tikzcd}
	{\Space^{A'}} & {\Space^{A}} \\
	{\Space^{B'}} & {\Space^{B}}
	\arrow["{f'_!}"', from=1-1, to=2-1]
	\arrow["\beta"{description}, Rightarrow, from=1-1, to=2-2]
	\arrow["{u^*}"', from=1-2, to=1-1]

із 2-морфізмом (2-cell) $\beta$ (морфізм у $\Space^A \to \Space^{B'}$), який є транспонуванням відображення:
\[\begin{tikzcd}
	{u^*} & {u^*f^*f_!} & {f'^* v^* f_!}
	\arrow["{u^* \eta}", Rightarrow, from=1-1, to=1-2]

Аналогічно до \ref{prop:action-postcomp}, ми даємо описання відображення Бека--Шевальє, як у \cite[Розділ 8.4]{cisinski:2024}.

Нехай $\DeclVar{F}{\GM}{A \to \Space}$, а $\varphi$ --- його корозпрямлення:

	\arrow[from=1-1, to=1-2]
	\arrow["\varphi"', from=1-1, to=2-1]

\begin{thm}[Smooth base change, \protect{\cite[Theorem 8.4.1]{cisinski:2024}}]
  Consider a pullback square
  \[\begin{tikzcd}
	{A'} & A \\
	{B'} & B
	\arrow["u", from=1-1, to=1-2]
	\arrow["{{f'}}"', from=1-1, to=2-1]
	\arrow["\lrcorner"{anchor=center, pos=0.125}, draw=none, from=1-1, to=2-2]
	\arrow["f", from=1-2, to=2-2]
	\arrow["v"', from=2-1, to=2-2]
\end{tikzcd}\]
малих категорій, де відображення $v$ є гладким. Тоді перетворення Бека--Шевальє є еквівалентністю
\[ f'_! u^* \simeq v^* f_!.\]
\end{thm}

\begin{proof}
Ми покажемо, що відображення $r$ із попереднього описання є еквівалентністю, довівши, що воно також є ініціальним (а за побудовою воно є коваріантним розшаруванням). Ми маємо наступний комутативний куб:
\[\begin{tikzcd}
	{A' \times_A X} && {B' \times_B Y} \\
	& X && Y \\
	{A'} && {B'} \\
	& A && B
	\arrow["{j''}"{description}, from=1-1, to=1-3]
	\arrow[from=1-1, to=2-2]
	\arrow["{u^*\phi}"{description, pos=0.3}, from=1-1, to=3-1]
	\arrow["\lrcorner"{anchor=center, pos=0.125}, draw=none, from=1-1, to=4-2]
	\arrow["{v'}"{description},from=1-3, to=2-4]
	\arrow["{v^*\psi}"{description, pos=0.3}, from=1-3, to=3-3]
	\arrow["\lrcorner"{anchor=center, pos=0.125}, draw=none, from=1-3, to=4-4]
	\arrow["j"{description, pos=0.4}, from=2-2, to=2-4, crossing over]
	\arrow["\psi"{description, pos=0.3}, from=2-4, to=4-4]
	\arrow["{f'}"{description, pos=0.7}, from=3-1, to=3-3]
	\arrow["u"{description}, from=3-1, to=4-2]
	\arrow["v"{description}, from=3-3, to=4-4]
	\arrow["f"{description, pos=0.6}, from=4-2, to=4-4]
  \arrow["\phi"{description, pos=0.3}, from=2-2, to=4-2, crossing over]
  \arrow["\lrcorner"{anchor=center, pos=0.125}, draw=none, from=3-1, to=4-4]
\end{tikzcd}\]
За лемою про декартові квадрати (pullback lemma), верхня грань також є декартовим квадратом. Оскільки $v$ є гладким, а відображення $j$ є ініціальним, відображення $j''$ є ініціальним. Проте $j'' = r \circ j'$, тому відображення $r$ також є ініціальним в силу скорочення (cancelation).
\end{proof}

Можна також довести аналогічну формулу \emph{правильної} зміни бази, як у \cite[Теорема 8.4.6]{cisinski:2024}.
\section{Висновки та майбутня робота}
\label{sec:conc}

Ми ввели та дослідили вплив $\infty$-категорного вкладення Йонеди в \STT{}. Це включає розробку класичних понять (розширення Кана, спряжені відображення, (ко)границі тощо) — і все це у синтетичному $\infty$-категорному контексті. Хоча деякі основи теорії вже досліджувалися в \STT{} раніше, нам вдалося отримати перші нетривіальні конкретні приклади, наприклад, спряжень (Теорема~\ref{thm:adj:precomp-is-right-adj}), та надати кілька більш витончених версій уже відомих теорем (Теорема~\ref{thm:tw:nat-iso}), які ближче відповідають їхнім стандартним аналогам.

\subsection{Споріднені роботи}

Існує кілька тісно пов'язаних тип-теоретичних підходів до синтетичної ($\infty$-)теорії категорій. Ми можемо грубо розділити їх на: (1) спрямовану теорію типів (directed type theory), де кожен тип є категорією, але різні операції ($\Prod{}$) мають бути обмежені, та (2) варіації на тему симпліційної теорії типів. Наприклад, протягом останніх років було запропоновано та досліджено багато спрямованих теорій типів~\parencite{%
  licata:2011,%
  warren:2013,%
  nuyts:2015,%
  kavvos:directed:2019,%
  buchholtz:2019,%
  kolomatskaia:2023,%
  weaver:2020,%
  weaver:phd,%
  ahrens:2023,%
  north:2018,%
  nuyts:2020,%
  neumann:2024,%
  }.
Загалом, хоча ці теорії типів є перспективним підходом до формалізації теорії категорій у теорії типів, жодна з них досі не привернула стільки уваги, скільки \STT{}, і, як наслідок, у жодній з них теорія категорій не була розвинена до такої міри, як у цій роботі. Крім того, розробити спрямовану теорію типів у такому стилі є значно складнішим завданням (оскільки це вимагає більш радикальної зміни базових правил теорії типів), і більшість пропозицій стосуються лише теорії $1$-категорій, а не $\prn{\infty,1}$-категорій. Проте ми зазначаємо, що деякі з цих теорій типів містять версію Теореми~\ref{thm:tw:strong-yoneda} у формі спрямованої індукції шляхів (directed path induction)~\parencite{north:2018,nuyts:2020,neumann:2024}. З огляду на те, що лише небагато наших аргументів спираються на типи, які не є категоріями, ми очікуємо, що багато з них перенесуться на достатньо багаті майбутні варіанти спрямованої теорії типів.

В літературі також розглядалися інші варіації симпліційної теорії типів. Наприклад, у кількох роботах використовується додаткова структура суджень (типи розширень, extension types), щоб отримати більше дефініційних рівностей навколо hom-типів~\parencite{riehl:2017,bardomiano:2021,weinberger:phd,buchholtz:2023,weinberger:twosided:2024,weinberger:sums:2024} ціною перетворення інтервалу на тип другого класу, подібного до дволівенової теорії типів~\parencite{annenkov:2023,voevodsky:2013}. Інші версії віддали перевагу кубічному інтервалу~\parencite{gratzer:2024} або навіть кубічному інтервалу поверх кубічної версії \HoTT{}~\parencite{weaver:2020,weaver:phd}. Окрім додавання модальностей, наша версія \STT{} є свідомо мінімалістичною: ми використовуємо лише звичайну \HoTT{} з невеликою кількістю постулатів. Відповідно, наші результати можуть бути інтерпретовані в практично будь-кому втіленні модальної \STT{} і не покладаються на додаткові дефініційні рівності.

Нарешті, існує багато спроб сформулювати більш концептуальні та синтетичні основи для теорії $\infty$-категорій, які не спираються на теорію типів. Наприклад, програма $\infty$-космосу (the $\infty$-cosmos program) \textcite{riehl:2022} має на меті дати систематичний опис формальної теорії категорій та незалежності від моделей, використовуючи 2-теорію категорій. З іншого боку, більшість практиків у цій галузі намагаються наводити більш вільні «незалежні від моделей» аргументи, уникаючи, наскільки це можливо, явних обчислень. Ми успішно перенесли деякі з цих аргументів у нашу структуру, довівши, що ця неформальна дисципліна є ефективною (наприклад, Розділ~\ref{sec:kan}). Нещодавно \textcite{cisinski:2024} розпочали переробку теорії $\infty$-категорій на свідомо неформальній мові високого рівня, знаходячи компроміс між формальною теорією на кшталт \STT{} та звичайною «незалежною від моделей» практикою. Ми очікуємо, що їхні аргументи можуть бути перекладені в \STT{}, і ми показали, що деякі з їхніх вихідних аксіом є \emph{довідними} в \STT{} (наприклад, Теореми~\ref{thm:prelims:space} та~\ref{thm:prelims:full-subcat}, а також Лема~\ref{lem:adj:functoriality-of-init-objects}).

\subsection{Майбутня робота}

Багато перспективних напрямків для майбутньої роботи залишаються відкритими для дослідження. Хоча ми зосередилися на категоріях передпучків та безпосередніх наслідках їхньої теорії, ми плануємо перенести інші фундаментальні результати теорії категорій (презентовані та доступні категорії, локалізації Бусфілда, теорію топосів тощо) в \STT{}. Також було б бажано адаптувати більше частин внутрішньої теорії $\infty$-категорій та теорії $\infty$-топосів Мартіні та Вольфа~\parencite{%
  martini:cocartesianfibrations:2022,%
  martini:colimitscocompletions:2021,%
  martini:phd,%
  martini:presentable:2022,%
  martini:propermorphisms:2023,%
  martini:yonedaslemma:2021,%
  wolf:phd%
  } до \STT{}. Крім того, ми сподіваємося розширити асистент доведення, такий як Agda~\parencite{agda}, необхідною підтримкою модальностей, щоб надати перевірені машиною версії доведень з цієї статті. З боку основ теорії, \STT{} наразі спирається на жменьку аксіом (Додаток~\ref{sec:axioms}) і тому задовольняє лише властивість нормалізації, але не канонічності. У майбутній роботі ми сподіваємося дослідити, яким із цих принципів можна надати обчислювальної інтерпретації та до якої міри можна «обчислювати» за допомогою синтетичних $\infty$-категорій.

Наступна \emph{аксіома двоїстості} була вперше досліджена в \textcite{blechschmidt:2023} і тягне за собою те,
що, наприклад, $\Int$ є категорією. Близько пов'язані аксіоми та наслідки розглядаються в \textcite{pugh:2025}
та \textcite{cherubini:2023}. Ми не вводили її в основній частині статті, оскільки вона не була явно використана
в жодному з наших доведень.

\begin{axiom}
  Якщо $A$ --- скінченно-породжена $\Int$-алгебра \textup(\ie{}, $A$ є обмеженою дистрибутивною решіткою,
еквівалентною фактор-кільцю $\Int[x_1,\ldots,x_n]$ за скінченною кількістю відношень\textup),
а $\Hom[\Int\mathrm{Alg}]{A}{\Int}$ --- тип гомоморфізмів $\Int$-алгебр, тоді відображення
$\lambda a\,f.\,f\prn{a} : A \to \prn{\Hom[\Int\mathrm{Alg}]{A}{\Int} \to \Int}$ є еквівалентністю.
\end{axiom}
