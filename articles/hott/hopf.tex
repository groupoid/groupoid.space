\documentclass{standalone}
\usepackage{tikz}
\usepackage{tikz-3dplot}
\usepackage{amsmath}
\begin{document}
\tdplotsetmaincoords{75}{120}
\begin{tikzpicture}[tdplot_main_coords, scale=1.0, line cap=round, line join=round]
\foreach \eta in {30,60,90,120} {
  \foreach \phi in {90,135,180,225,270} {
    \draw[very thin, black] plot[smooth, samples=100, domain=0:360, variable=\theta]
      (
        { sqrt((1 - sin(\phi - \theta) * cos(\eta/2))/(1 + sin(\phi - \theta) * cos(\eta/2))) * cos(\phi + \theta) * sin(\eta/2) /
          sqrt((cos(\phi + \theta) * sin(\eta/2))^2 + (sin(\phi + \theta) * sin(\eta/2))^2 + (cos(\phi - \theta) * cos(\eta/2))^2)},
        { sqrt((1 - sin(\phi - \theta) * cos(\eta/2))/(1 + sin(\phi - \theta) * cos(\eta/2))) * sin(\phi + \theta) * sin(\eta/2) /
          sqrt((cos(\phi + \theta) * sin(\eta/2))^2 + (sin(\phi + \theta) * sin(\eta/2))^2 + (cos(\phi - \theta) * cos(\eta/2))^2)},
        { sqrt((1 - sin(\phi - \theta) * cos(\eta/2))/(1 + sin(\phi - \theta) * cos(\eta/2))) * cos(\phi - \theta) * cos(\eta/2) /
          sqrt((cos(\phi + \theta) * sin(\eta/2))^2 + (sin(\phi + \theta) * sin(\eta/2))^2 + (cos(\phi - \theta) * cos(\eta/2))^2)}
      );
  }
}
\end{tikzpicture}
\end{document}
